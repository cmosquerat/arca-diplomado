\begin{frame}[fragile,shrink=28]{Máscaras Booleanas}
  \vspace{-6pt}
  {\footnotesize
    Una condición sobre una columna produce True/False.
    Al pasarla como índice, pandas filtra.}

  \vspace{4pt}
  \begin{pyblock}
mujeres = df[df["Sex"] == "female"]
print(f"Mujeres: {len(mujeres)} de {len(df)}")

primera_clase = df[df["Pclass"] == 1]
print(f"1ra clase: {len(primera_clase)}")

jovenes = df[df["Age"] < 10][["Sex", "Age", "Survived"]]
jovenes.head()
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ \texttt{df[condición]} $\to$ solo filas donde es \texttt{True}.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Múltiples Condiciones: \texttt{\&}, \texttt{|}, \texttt{\~{}}}
  \vspace{-6pt}
  {\scriptsize
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Operador} & \textbf{Significado} & \textbf{Python} \\
    \midrule
    \texttt{\&} & Y lógico & \texttt{and} \\
    \texttt{|}  & O lógico & \texttt{or} \\
    \texttt{\~{}} & NO lógico & \texttt{not} \\
    \bottomrule
  \end{tabular}
  }

  \vspace{4pt}
  \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
# & = Y (ambas verdaderas)
m_1ra = df[(df["Sex"] == "female") & (df["Pclass"] == 1)]
print(f"Mujeres 1ra clase: {len(m_1ra)}")

# | = O (al menos una)
ninos_o_1 = df[(df["Age"] < 12) | (df["Pclass"] == 1)]
print(f"Ninos O 1ra clase: {len(ninos_o_1)}")

# ~ = NO (invierte)
no_s = df[~(df["Embarked"] == "S")]
print(f"NO Southampton: {len(no_s)}")
  \end{pyblock}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
    \faExclamationTriangle\ Cada condición debe ir
    \textbf{entre paréntesis}.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Filtros Especiales: \texttt{isin()} y \texttt{between()}}
  \vspace{-6pt}
  \begin{pyblock}
# isin() -- como IN de SQL
puertos_cq = df[df["Embarked"].isin(["C", "Q"])]
print(f"Embarcaron en C o Q: {len(puertos_cq)}")

# between() -- rango inclusivo
adultos = df[df["Age"].between(30, 50)]
print(f"Edad 30-50: {len(adultos)}")

# Negacion con isin
solo_s = df[~df["Embarked"].isin(["C", "Q"])]
print(f"Solo Southampton: {len(solo_s)}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{isin(lista)} evita cadenas de \texttt{|}.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{between(a, b)} incluye ambos extremos.
  \end{itemize}
  }
\end{frame}

% =============================================================================

\sectionframe{\texttt{groupby()}}{Resumir por grupos}

\begin{frame}[fragile,shrink=28]{Anatomía de \texttt{groupby}}
  \vspace{-6pt}
  {\footnotesize
    Para preguntas tipo ``¿cuál es el promedio \textbf{por grupo}?''}

  \vspace{4pt}
  \begin{pyblock}
# Cuantos por sexo?
print(df.groupby("Sex")["PassengerId"].count())
print()
# Tarifa promedio por clase?
print(df.groupby("Pclass")["Fare"].mean().round(2))
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize
  \begin{tabular}{@{}ll@{}}
    \toprule
    \textbf{Función} & \textbf{Responde\ldots} \\
    \midrule
    \texttt{.count()} & ¿Cuántos? \\
    \texttt{.sum()} & ¿Cuánto en total? \\
    \texttt{.mean()} & ¿Cuál es el promedio? \\
    \texttt{.min() / .max()} & ¿Cuál es el extremo? \\
    \bottomrule
  \end{tabular}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{\texttt{groupby() + agg()} --- Varias Métricas}
  \vspace{-6pt}
  {\footnotesize
    En lugar de tres \texttt{groupby} separados,
    \texttt{.agg()} calcula todo de una vez:}

  \vspace{4pt}
  \begin{pyblock}
resumen = df.groupby("Pclass").agg(
    total       = ("PassengerId", "count"),
    sobreviven  = ("Survived",    "sum"),
    tarifa_prom = ("Fare",        "mean"),
).round(1)
print(resumen)
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Sintaxis: \texttt{nombre = ("columna", "funcion")}.
    \item[{\color{arcaRed}\scriptsize\faLightbulb}]
      Equivale a una tabla dinámica de Excel.
  \end{itemize}
  }
\end{frame}

    \toprule
    \textbf{Estrategia} & \textbf{Cuándo} \\
    \midrule
    \texttt{dropna()} & Pocos NaN ($<$ 5\%) \\
    \texttt{fillna(mediana)} & Columnas numéricas \\
    \texttt{fillna("valor")} & Columnas de texto \\
    \bottomrule
  \end{tabular}
  }

  \vspace{4pt}
  \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
# Numerica -> mediana
mediana = df["Age"].median()
df["Age"] = df["Age"].fillna(mediana)
print(f"Age NaN: {df['Age'].isna().sum()}")

# Categorica -> valor mas frecuente
df["Embarked"] = df["Embarked"].fillna("S")

# dropna -> eliminar filas
df2 = df.dropna(subset=["Fare"])
print(f"dropna: {len(df)} -> {len(df2)} filas")
  \end{pyblock}
\end{frame}
