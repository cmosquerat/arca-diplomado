% ─────────────────────────────────────────────────────────────────────────────
%  APERTURA: BASE 10 VS BASE 2
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[shrink=30]{Un número puede ser exacto en una base y no en otra}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    La idea clave es esta: \textbf{que un número se vea simple en base 10
    no significa que sea simple en base 2}.}

  \vspace{8pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\color{arcaDark}\bfseries En base 10:}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item \textbf{1/2 = 0.5} termina.
        \item \textbf{1/4 = 0.25} termina.
        \item \textbf{1/3 = 0.3333...} no termina nunca.
      \end{itemize}
      }

      \vspace{6pt}
      {\footnotesize\color{arcaDark}
        En decimal eso nos parece normal:
        algunas fracciones terminan y otras no.}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\color{arcaDark}\bfseries En base 2 (binario):}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item \textbf{1/2 = 0.1 en base 2} sí termina.
        \item \textbf{1/4 = 0.01 en base 2} sí termina.
        \item \textbf{0.1 en base 10 = 0.0001100110011... en base 2} no termina.
      \end{itemize}
      }

      \vspace{6pt}
      {\footnotesize\color{arcaDark}
        O sea: \texttt{0.1} y \texttt{0.3} son números cortos en decimal,
        pero en binario se vuelven secuencias infinitas.
        Por ejemplo, \texttt{0.3 en base 10 = 0.0100110011... en base 2}.}
    \end{column}
  \end{columns}

  \vspace{8pt}
  {\footnotesize\color{textMuted}
    Igual que \texttt{1/3} obliga a escribir \texttt{0.3333...} en decimal,
    \texttt{0.1} obliga a escribir una secuencia infinita en binario.
    Como la máquina no puede guardar infinitos bits, guarda una
    \textbf{aproximación}.}
\end{frame}

\begin{frame}[fragile,shrink=25]{¿Por qué \texttt{0.1 + 0.2} no da exactamente \texttt{0.3}?}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{pyblock}
print(0.1 + 0.2)
print(0.1 + 0.2 == 0.3)

# Ver la representacion interna
print(format(0.1, ".50f"))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
0.30000000000000004
False
0.1000000000000000055511151231257827...
      \end{outputblock}

      \vspace{4pt}
      {\scriptsize\color{textMuted}
      Ni \texttt{0.1} ni \texttt{0.2} se guardan exactos.
      Al sumar, los pequeños errores se acumulan.}
    \end{column}

    \begin{column}{0.48\textwidth}
      \vspace{2pt}
      {\footnotesize\color{arcaDark}\bfseries En la práctica:}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item Para \textbf{mostrar}: \texttt{round(0.1+0.2, 2)} $\rightarrow$ \texttt{0.3}.
        \item Para \textbf{comparar}: no usar \texttt{==}; usar
          \texttt{abs(a - b) < 1e-9}.
        \item Para \textbf{dinero}: módulo \texttt{decimal} o
          enteros (centavos).
      \end{itemize}
      }

      \vspace{6pt}
      {\footnotesize\color{arcaDark}\bfseries Conclusión:}
      {\scriptsize\color{textMuted}
      Es el comportamiento normal del punto flotante en cualquier lenguaje.
      Saber que ``0.1 en decimal no es entero en binario'' evita sorpresas.}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}[fragile,shrink=30]{Acceso Seguro con \texttt{.get()}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    \mintinline{python}{dict["clave"]} da error si la clave no existe.
    \texttt{.get()} lo evita.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{pyblock}
producto = {
    "nombre": "Coca-Cola 600ml",
    "precio": 1.50,
    "stock": 200
}

print(producto["nombre"])
print(producto.get("precio"))
print(producto.get("color", "N/A"))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Coca-Cola 600ml
1.5
N/A
      \end{outputblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \mintinline{python}{dict["clave"]} accede directamente.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \mintinline{python}{.get("clave")} hace lo mismo
          pero no da error si no existe.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \mintinline{python}{.get("clave", "N/A")} devuelve
          \mintinline{python}{"N/A"} como valor por defecto.
        \item[{\color{arcaRed}\scriptsize\faExclamationTriangle}]
          \mintinline{python}{producto["color"]} daría
          \mintinline{python}{KeyError} porque \mintinline{python}{"color"}
          no existe.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

          \textbf{modifica} si la clave ya existe.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \mintinline{python}{dict["nueva"] = valor}

          la clave y su valor.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Es la misma sintaxis para modificar
          y agregar. Python decide según si
          la clave ya está o no.
      \end{itemize}
      }

      \vspace{-2pt}
      \begin{outputblock}
Guardado: 375.0
Con IVA: 420.0
      \end{outputblock}
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\footnotesize\color{arcaDark}
    \texttt{print()} muestra texto en pantalla.
    \texttt{return} entrega un valor que el programa puede
    guardar, operar o pasar a otra función.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Parámetros con Valor por Defecto}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \vspace{2pt}
      \begin{pyblock}
def calcular_ingreso(cajas, precio=12.50):
    return cajas * precio

print(calcular_ingreso(30))
print(calcular_ingreso(30, 15.00))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
375.0
450.0
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{precio=12.50} es un valor por defecto.
          Si no se pasa, usa 12.50.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Los parámetros con default van \textbf{al final}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Si llamo \texttt{calcular\_ingreso(30)},
          Python usa 12.50 automáticamente.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Si llamo \texttt{calcular\_ingreso(30, 15.00)},
          Python reemplaza el valor por defecto.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Los valores por defecto son útiles cuando
          una opción se repite mucho y solo a veces cambia.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=28]{Alcance de Variables (Scope)}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      {\footnotesize\color{arcaDark}\bfseries Variable local dentro de la función:}
      \vspace{2pt}
      \begin{pyblock}
def duplicar(x):
    resultado = x * 2
    return resultado

print(duplicar(5))
# print(resultado)  # Error: no existe aqui
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
10
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Las variables creadas \textbf{dentro}
          de una función son \textbf{locales}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Eso significa que solo existen mientras
          la función se ejecuta.
        \item[{\color{arcaRed}\scriptsize\faExclamationTriangle}]
          Intentar acceder a \texttt{resultado}
          fuera de \texttt{duplicar()} da
          \texttt{NameError}.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Si quieren usar ese valor afuera,
          deben sacarlo con \texttt{return}
          y guardarlo en otra variable.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{¿Una Función Puede Cambiar Algo Fuera de su Scope?}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Sí, pero depende de \textbf{qué recibe} y de \textbf{qué hace} con eso.}

  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Reasignar una variable local no cambia la de afuera}
      \begin{pyblock}
def cambiar_numero(x):
    x = x + 10

valor = 5
cambiar_numero(valor)
print(valor)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
5
      \end{outputblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Modificar una lista o diccionario sí se refleja afuera}
      \begin{pyblock}
def agregar_alerta(alertas):
    alertas.append("PM2.5 alto")

estado = []
agregar_alerta(estado)
print(estado)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
estado = ['PM2.5 alto']
      \end{outputblock}
    \end{column}
  \end{columns}

  \vspace{1pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Reasignar una variable dentro de la función
      no cambia la variable externa.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Si reciben una \textbf{lista} o \textbf{diccionario}
      y lo modifican, el cambio sí se ve afuera.
    \item[{\color{arcaRed}\scriptsize\faLightbulb}]
      Regla práctica: para evitar confusión,
      prefieran \textbf{\texttt{return}}.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=30]{Funciones que Procesan Colecciones}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Las funciones pueden recibir listas o diccionarios como parámetros.}

  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
def resumen_ventas(ventas, meta):
    total = 0
    sobre_meta = 0
    for v in ventas:
        total += v
        if v >= meta:
            sobre_meta += 1
    return total, sobre_meta

datos = [150, 80, 200, 45, 310, 60]
t, s = resumen_ventas(datos, 100)
print(f"Total: {t}")
print(f"Sobre meta: {s}")
      \end{pyblock}
    \end{column}

    \begin{column}{0.44\textwidth}

Total: 845
Sobre meta: 3
      \end{outputblock}

      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Se pueden retornar \textbf{varios valores}
          separados por coma.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Al llamar: \texttt{t, s = func(...)}
          desempacan los 2 valores.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          La lógica de acumuladores y contadores
          que ya dominan ahora vive
          dentro de una función reutilizable.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}
