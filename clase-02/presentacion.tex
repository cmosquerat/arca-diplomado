\begin{frame}[fragile,shrink=30]{¿Qué es un Booleano?}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Es un tipo de dato que solo puede tener dos valores:
    \texttt{True} (verdadero) o \texttt{False} (falso).
    Es la base para que el programa pueda \textbf{decidir}.}

  \vspace{8pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\color{arcaDark}\bfseries
        En la Clase 1 vimos \texttt{int}, \texttt{float}, \texttt{str}.
        El booleano (\texttt{bool}) responde preguntas de sí/no.}

      \vspace{8pt}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{6pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          ``¿Hay stock suficiente?''
          $\rightarrow$ \texttt{True} o \texttt{False}

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          ``¿El cliente tiene crédito?''
          $\rightarrow$ \texttt{True} o \texttt{False}

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          El programa usa ese resultado
          para elegir qué camino seguir.
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{pyblock}
stock = 150
pedido = 200
hay_suficiente = stock >= pedido
print(hay_suficiente)
print(type(hay_suficiente))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
False
<class 'bool'>
      \end{outputblock}

      \vspace{6pt}
      {\scriptsize\color{textMuted}
        Las comparaciones devuelven booleanos.
        A partir de ahí, el programa toma decisiones.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Operadores de Comparación}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Comparan dos valores y devuelven \texttt{True} o \texttt{False}.}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      {\footnotesize
      \renewcommand{\arraystretch}{1.35}
      \begin{tabular}{@{}c l l@{}}
        \toprule
        {\bfseries\color{arcaDark} Op.}
        & {\bfseries\color{arcaDark} Significado}
        & {\bfseries\color{arcaDark} Ejemplo} \\
        \midrule
        \texttt{==} & igual a              & \texttt{stock == 150} \\
        \texttt{!=} & distinto de          & \texttt{ruta != "Sur"} \\
        \texttt{>}  & mayor que            & \texttt{margen > 0.20} \\
        \texttt{<}  & menor que            & \texttt{stock < minimo} \\
        \texttt{>=} & mayor o igual        & \texttt{precio >= 1.50} \\
        \texttt{<=} & menor o igual        & \texttt{dias <= 30} \\
        \bottomrule
      \end{tabular}
      }

      \vspace{6pt}
      {\scriptsize\color{arcaRed}\bfseries
        \faExclamationTriangle\hspace{4pt}
        \texttt{==} compara. \texttt{=} asigna.
        No confundirlos.}
    \end{column}

    \begin{column}{0.46\textwidth}
      \begin{pyblock}
stock  = 150
pedido = 200
minimo = 100
precio = 1.50
print(stock >= pedido)
print(stock >= minimo)
print(precio == 1.50)
print(precio != 1.25)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
False
True
True
True
      \end{outputblock}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=25]{Operadores Lógicos}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Combinan condiciones. Pensémoslos con analogías:}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize
      \begin{itemize}\setlength{\itemsep}{6pt}
        \item[{\color{arcaRed}\bfseries and}]
          \textbf{Candado con dos llaves}\\
          {\scriptsize\color{textMuted}
            Necesitas la llave A \emph{y} la B.
            Si falta una, no abre.}
          \\[2pt]
          $\Rightarrow$ \texttt{True} solo si \emph{ambos} son \texttt{True}.

        \item[{\color{arcaRed}\bfseries or}]
          \textbf{¿Hay café o té?}\\
          {\scriptsize\color{textMuted}
            Con que exista uno alcanza.
            No necesitas los dos.}
          \\[2pt]
          $\Rightarrow$ \texttt{True} si \emph{al menos uno} es \texttt{True}.

        \item[{\color{arcaRed}\bfseries not}]
          \textbf{El interruptor}\\
          {\scriptsize\color{textMuted}
            Enciende lo apagado, apaga lo encendido.
            Invierte el estado.}
          \\[2pt]
          $\Rightarrow$ \texttt{True} $\leftrightarrow$ \texttt{False}.
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{pyblock}
hay_stock         = True
tiene_credito     = False
cliente_bloqueado = False
print(hay_stock and tiene_credito)
print(hay_stock or tiene_credito)
print(not cliente_bloqueado)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
False
True
True
      \end{outputblock}

      \vspace{4pt}
      {\scriptsize\color{textMuted}
        Despacho: stock \textbf{and} crédito.
        Alerta: falta uno \textbf{or} otro.
        \textbf{not} bloqueado = puede comprar.}
    \end{column}
  \end{columns}
\end{frame}

    \item[{\color{arcaRed}\bfseries 3.}]
      ¿\textbf{Todos} los productos tienen stock suficiente? (\texttt{and})
  \end{enumerate}
  }

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb[regular]\hspace{4pt}\textbf{Pista:}
    \texttt{alerta\_coca = stock\_coca < minimo\_coca}}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 8: SOLUCIÓN — ALERTAS DE STOCK

\begin{frame}[fragile,shrink=30]{Solución --- Alertas de Stock}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \begin{pyblock}
stock_coca  = 150; minimo_coca  = 200
stock_sprite = 45; minimo_sprite = 50
stock_dasani = 320; minimo_dasani = 100
alerta_coca   = stock_coca < minimo_coca

todo_ok = not alerta_coca and not alerta_sprite and not alerta_dasani

          Coca-Cola y Sprite necesitan reposición.
        \item[{\color{arcaRed}\scriptsize\faCheck}]

        \item[{\color{arcaRed}\scriptsize\faCheck}]

      }

%  FRAME 8b: PRÁCTICA — ¿SE ACTIVA LA ALARMA?

\begin{frame}[fragile,shrink=30]{Práctica: ¿Se Activa la Alarma? \hfill {\normalsize\color{arcaRed}\faClock\hspace{4pt}7 min}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Una bodega tiene sensores de temperatura y humedad.
    La alarma se activa si hay riesgo ambiental
    \textbf{y} el sistema no está en mantenimiento.}

  \vspace{5pt}
  \begin{pyblock}
temperatura = 38      # grados Celsius
humedad     = 72      # porcentaje
en_mantenimiento = False
  \end{pyblock}

  \vspace{4pt}
  {\footnotesize\color{textMuted}
    Hay \textbf{riesgo ambiental} si la temperatura
    supera 35°C \textbf{o} la humedad supera 80\%.
    Pero si el sistema está en mantenimiento, la alarma
    \textbf{no} debe sonar (para evitar falsas alertas).
  }

  \vspace{5pt}
  {\footnotesize
  \begin{enumerate}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\bfseries 1.}]
      Construyan la variable \texttt{riesgo} usando \texttt{or}.
    \item[{\color{arcaRed}\bfseries 2.}]
      Construyan \texttt{alarma} combinando \texttt{riesgo}
      con \texttt{not en\_mantenimiento} usando \texttt{and}.
    \item[{\color{arcaRed}\bfseries 3.}]
      ¿Qué pasa si cambian \texttt{en\_mantenimiento = True}?
      ¿Y si ponen \texttt{temperatura = 30} y \texttt{humedad = 60}?
  \end{enumerate}
  }

  \vspace{3pt}
  {\scriptsize\color{arcaRed}\bfseries
    \faExclamationTriangle\hspace{4pt}Ojo:
    si usan \texttt{and} en vez de \texttt{or} para el riesgo,
    la alarma casi nunca sonaría. ¿Por qué?}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 8c: SOLUCIÓN — ¿SE ACTIVA LA ALARMA?
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{Solución --- ¿Se Activa la Alarma?}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
temperatura = 38
humedad     = 72
en_mantenimiento = False

riesgo = temperatura > 35 or humedad > 80
alarma = riesgo and not en_mantenimiento

print("Riesgo:", riesgo)
print("Alarma:", alarma)

# Mantenimiento activo
en_mantenimiento = True
alarma = riesgo and not en_mantenimiento
print("Con mantenimiento:", alarma)

# Sin riesgo real
riesgo2 = 30 > 35 or 60 > 80
print("Sin riesgo:", riesgo2)
      \end{pyblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{4pt}
      \begin{outputblock}
Riesgo: True
Alarma: True
Con mantenimiento: False
Sin riesgo: False
      \end{outputblock}

      \vspace{4pt}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{or}: basta que \textbf{uno}
          de los sensores detecte peligro.
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{and not}: el riesgo existe,
          pero el mantenimiento lo silencia.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Con \texttt{and} en vez de \texttt{or},
          necesitaríamos calor \textbf{y} humedad
          \textbf{al mismo tiempo} para sonar.
          Eso es un error lógico común.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 9: ¿QUÉ ES UN CONDICIONAL?
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[shrink=10]{¿Qué es un Condicional?}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Los booleanos nos dicen verdadero/falso.
    Los condicionales nos dicen \textbf{qué hacer} según ese resultado.}

  \vspace{8pt}
  \centering
  \begin{tikzpicture}[
    node distance=0.6cm,
    sbox/.style={rectangle, rounded corners=4pt,
      minimum width=3.4cm, minimum height=0.65cm,
      font=\scriptsize\bfseries, text=white, align=center},
    flowdiam/.style={diamond, draw=arcaRed, fill=arcaRed!10,
      minimum width=3.2cm, minimum height=0.9cm,
      font=\scriptsize\color{arcaDark}\bfseries, align=center,
      inner sep=0pt},
    sarrow/.style={-{Stealth[length=4pt]}, line width=0.8pt, arcaRed}
  ]
    \node[sbox, fill=arcaDark] (inicio) {Margen calculado};
    \node[flowdiam, below=0.5cm of inicio] (cond) {margen >= 0.20?};
    \node[sbox, fill=codeGreen, below left=0.7cm and 1.2cm of cond]
      (si) {\faCheck\hspace{4pt}Aprobar ruta};
    \node[sbox, fill=arcaRed, below right=0.7cm and 1.2cm of cond]
      (no) {\faTimes\hspace{4pt}Revisar ruta};

    \draw[sarrow] (inicio) -- (cond);
    \draw[sarrow] (cond) -- node[left, font=\tiny\color{codeGreen}]
      {\textbf{True}} (si);
    \draw[sarrow] (cond) -- node[right, font=\tiny\color{arcaRed}]
      {\textbf{False}} (no);
  \end{tikzpicture}

  \vspace{6pt}
  {\footnotesize\color{textMuted}
    El programa evalúa la condición y toma un camino diferente
    según el resultado. Eso es \textbf{control de flujo}.}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 10: if / else
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{if / else — Dos Caminos}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Si la condición es verdadera, ejecutamos un bloque;
    si no, el otro.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      {\footnotesize\color{arcaDark}\bfseries Estructura:}
      \vspace{2pt}
      \begin{tcolorbox}[
        colback=arcaGray, colframe=arcaRed, boxrule=0pt, leftrule=3pt,
        arc=2pt, left=5pt, right=5pt, top=3pt, bottom=3pt
      ]
        {\scriptsize\ttfamily\color{textDark}
          \textbf{\color{arcaRed}if} condicion:\\ \hspace*{12pt}bloque\_si\_true\\
          \textbf{\color{arcaRed}else}:\\ \hspace*{12pt}bloque\_si\_false
        }
      \end{tcolorbox}

      \vspace{4pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item[{\color{arcaRed}\scriptsize\faIndent}]
          La \textbf{indentación} (4 espacios) define
          qué pertenece a cada bloque.
        \item[{\color{arcaRed}\scriptsize\textbf{:}}]
          Los \textbf{dos puntos} \texttt{:} al final
          de la condición son obligatorios.
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.46\textwidth}
      \begin{pyblock}
stock = 45
minimo = 50
if stock < minimo:
    print("ALERTA: reabastecer")
else:
    print("Stock suficiente")
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
ALERTA: reabastecer
      \end{outputblock}

      \vspace{6pt}
      {\scriptsize\color{textMuted}
        \faArrowRight\hspace{4pt}
        \texttt{stock} es 45, menor que 50: entra al bloque \texttt{if}.
        Si cambiamos a 60, entraría al \texttt{else}.}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 11: if / elif / else
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{if / elif / else — Múltiples Caminos}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Cuando hay más de dos posibilidades, usamos \texttt{elif}
    para encadenar condiciones.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
margen = 0.18
if margen >= 0.25:
    clasificacion = "Rentable"
elif margen >= 0.15:
    clasificacion = "En riesgo"
else:
    clasificacion = "Critica"
print("Clasificacion:", clasificacion)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Clasificacion: En riesgo
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Python evalúa las condiciones
          \textbf{de arriba hacia abajo}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          En cuanto una es \texttt{True},
          ejecuta ese bloque y \textbf{sale}.
          No sigue evaluando.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{else} es el \textbf{caso por defecto}:
          se ejecuta si ninguna condición
          fue verdadera.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{elif} es contracción de
          ``else if''.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 14: ¿QUÉ ES UNA LISTA?
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{¿Qué es una Lista?}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Hasta ahora, una variable guarda un solo valor.
    Una \textbf{lista} guarda varios bajo el mismo nombre.}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimes\hspace{4pt}
        Sin lista — 4 variables separadas}
      \begin{pyblock}
ruta1 = "Guayaquil Norte"
ruta2 = "Guayaquil Sur"
ruta3 = "Manta"
ruta4 = "Cuenca"
      \end{pyblock}

      \vspace{6pt}
      {\scriptsize\color{codeGreen}\bfseries\faCheck\hspace{4pt}
        Con lista — un solo nombre}
      \begin{pyblock}
rutas = ["Guayaquil Norte", "Guayaquil Sur",
         "Manta", "Cuenca"]
print(len(rutas))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
4
      \end{outputblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Los elementos van entre \textbf{corchetes}
          \texttt{[]} separados por comas.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Puede contener \textbf{cualquier tipo}:
          números, texto, booleanos,
          incluso mezclarlos.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{len(lista)} devuelve el
          número de elementos.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Si añadimos una ruta, \textbf{una sola
          línea} cambia, no cuatro variables.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 15: ACCEDER Y MODIFICAR ELEMENTOS

\begin{frame}[fragile,shrink=30]{Acceder y Modificar Elementos}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Cada elemento tiene un \textbf{índice} que empieza en \textbf{cero}.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
precios = [1.50, 1.25, 0.90, 1.75]
print(precios[0])
print(precios[3])
print(precios[-1])
print(precios[1:3])
precios[0] = 1.65
print(precios[0])
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
1.5
1.75
1.75
[1.25, 0.9]
1.65
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \centering
      \vspace{4pt}
      \begin{tikzpicture}[
        cel/.style={draw=arcaRed, fill=arcaRed!8, minimum width=1.1cm,
                    minimum height=0.65cm, font=\scriptsize\ttfamily},
        idx/.style={font=\tiny\color{textMuted}},

      ]
        \node[cel] (c0) {1.50};

        \node[cel, right=of c1] (c2) {0.90};
        \node[cel, right=of c2] (c3) {1.75};

        \node[idx, below=3pt of c0] {0};
        \node[idx, below=3pt of c1] {1};
        \node[idx, below=3pt of c2] {2};
        \node[idx, below=3pt of c3] {3};

        \node[idx, above=3pt of c0] {-4};
        \node[idx, above=3pt of c1] {-3};
        \node[idx, above=3pt of c2] {-2};
        \node[idx, above=3pt of c3] {-1};
      \end{tikzpicture}

      \vspace{10pt}
      {\scriptsize\color{arcaRed}\bfseries
        \faExclamationTriangle\hspace{4pt}
        El primer elemento es el índice \textbf{0},
        no el 1.}

      \vspace{6pt}
      {\scriptsize\color{textMuted}

  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Métodos de Lista}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Las listas tienen \textbf{métodos} propios: instrucciones que actúan
    directamente sobre la lista.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
rutas = ["Guayaquil Norte", "Manta"]
rutas.append("Cuenca")
print(rutas)
print(len(rutas))
rutas.sort()
print(rutas)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}[]
['Guayaquil Norte', 'Manta', 'Cuenca']
3
['Cuenca', 'Guayaquil Norte', 'Manta']
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      {\footnotesize
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{@{}l l@{}}
        \toprule
        {\bfseries\color{arcaDark} Método}
        & {\bfseries\color{arcaDark} Qué hace} \\
        \midrule
        \texttt{.append(x)} & Agrega \texttt{x} al final \\
        \texttt{.sort()}    & Ordena en el lugar \\
        \texttt{.remove(x)} & Elimina la primera \texttt{x} \\
        \texttt{.pop(i)}    & Elimina el índice \texttt{i} \\
        \bottomrule
      \end{tabular}
      }

      \vspace{6pt}
      {\scriptsize\color{textMuted}
        La diferencia con funciones como \texttt{len()}
        es que los métodos van
        \textbf{después del punto}: \texttt{lista.metodo()}.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{¿Notaron el Problema?}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Con 8 pedidos ya es incómodo. ¿Qué hacemos con 800?}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Sin ciclos --- impracticable}
      \begin{pyblock}
print("Pedido 1:", pedidos[0])
print("Pedido 2:", pedidos[1])
print("Pedido 3:", pedidos[2])
print("...")
      \end{pyblock}
    \end{column}

    \begin{column}{0.50\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Con ciclo \texttt{for} --- funciona para 8 o 8\,000}
      \begin{pyblock}
pedidos = [120, 45, 200, 30, 180, 95, 60, 75]
for cantidad in pedidos:
    print("Pedido:", cantidad)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Pedido: 120
Pedido: 45
Pedido: 200
...
      \end{outputblock}
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\footnotesize\color{arcaDark}
    \textbf{El ciclo \texttt{for}} toma cada elemento de la lista,
    lo asigna a una variable temporal (\texttt{cantidad}),
    y ejecuta el bloque indentado con ese valor.}
\end{frame}

\begin{frame}[fragile,shrink=30]{Ciclo \texttt{for} con \texttt{range()}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Cuando necesitamos contar (días, meses, posiciones),
    usamos \texttt{range()}.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{pyblock}
for dia in range(5):
    print("Dia", dia + 1)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Dia 1
Dia 2
...
Dia 5
      \end{outputblock}

      \vspace{2pt}
      \begin{pyblock}
rutas    = ["GYE Norte", "Manta", "Cuenca"]
ingresos = [15200, 12400, 6300]
for i in range(len(rutas)):
    print(rutas[i], "->", ingresos[i])
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
GYE Norte -> 15200
Manta -> 12400
Cuenca -> 6300
      \end{outputblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{range(n)} genera los números
          \textbf{0 hasta n-1}.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{range(len(lista))} genera los
          índices válidos de la lista:
          0, 1, 2, \ldots

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Al acceder por índice \texttt{i},
          podemos sincronizar
          \textbf{dos listas en paralelo}.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Útil para calcular métricas
          que relacionan varios atributos
          del mismo elemento.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Entendiendo \texttt{range(inicio, fin, paso)}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    \texttt{range()} puede recibir 1, 2 o 3 argumentos.}

  \vspace{5pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      {\footnotesize
      \renewcommand{\arraystretch}{1.35}
      \begin{tabular}{@{}l l@{}}
        \toprule
        {\bfseries\color{arcaDark} Forma}
        & {\bfseries\color{arcaDark} Qué significa} \\
        \midrule
        \texttt{range(fin)} &
        Empieza en 0, avanza de 1 en 1 \\
        \texttt{range(inicio, fin)} &
        Empieza en \texttt{inicio}, paso 1 \\
        \texttt{range(inicio, fin, paso)} &
        Empieza en \texttt{inicio}, avanza de \texttt{paso} \\
        \bottomrule
      \end{tabular}
      }

      \vspace{7pt}
      {\scriptsize\color{arcaRed}\bfseries
        \faExclamationTriangle\hspace{4pt}
        El valor \texttt{fin} nunca se incluye.}
    \end{column}

    \begin{column}{0.44\textwidth}
      \begin{pyblock}
print(list(range(5)))
print(list(range(2, 7)))
print(list(range(10, 0, -2)))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
range(5): [0, 1, 2, 3, 4]
range(2, 7): [2, 3, 4, 5, 6]
range(10, 0, -2): [10, 8, 6, 4, 2]
      \end{outputblock}

      \vspace{5pt}
      {\scriptsize\color{textMuted}
        Si el paso es negativo, la secuencia
        va hacia atrás.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{El Patrón Acumulador}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Para sumar, contar o acumular sobre una lista,
    iniciamos una variable en cero y la vamos sumando.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
pedidos = [120, 45, 200, 30, 180, 95, 60, 75]
total = 0
for cantidad in pedidos:
    total = total + cantidad
print("Total de cajas:", total)

pedidos_grandes = 0
for cantidad in pedidos:
    if cantidad >= 100:
        pedidos_grandes = pedidos_grandes + 1
print("Pedidos >= 100:", pedidos_grandes)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Total de cajas: 805
Pedidos >= 100: 3
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{6pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \textbf{Antes del ciclo:}
          inicializar el acumulador en cero.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \textbf{Dentro del ciclo:}
          sumar el valor actual al
          acumulador.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \textbf{Después del ciclo:}
          el acumulador tiene el total.

        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          \texttt{total = total + cantidad}
          se puede abreviar como
          \texttt{total += cantidad}.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  FRAME 22b: CONTADORES
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{Contadores --- Contar Eventos en un Ciclo}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Un \textbf{contador} es un acumulador especial: suma \texttt{1}
    cada vez que ocurre algo.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
ventas = [150, 80, 200, 45, 310, 60, 95]
meta = 100

cumplieron = 0
no_cumplieron = 0

for venta in ventas:
    if venta >= meta:
        cumplieron += 1
    else:
        no_cumplieron += 1

print("Cumplieron:", cumplieron)
print("No cumplieron:", no_cumplieron)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Cumplieron: 3
No cumplieron: 4
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{2pt}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \textbf{Acumulador:} suma \textbf{valores}
          (\texttt{total += venta}).

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \textbf{Contador:} suma \textbf{1}
          (\texttt{cont += 1}) cuando una
          condición se cumple.

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Se pueden usar \textbf{varios contadores}
          en un mismo ciclo para clasificar.

        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Siempre inicializar en \texttt{0}
          \textbf{antes} del \texttt{for}.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{\texttt{for} + \texttt{if} — La Combinación Poderosa}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    El \texttt{for} recorre; el \texttt{if} filtra.
    Juntos nos permiten procesar \textbf{y} decidir.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{pyblock}
pedidos = [120, 45, 200, 30, 180]
minimo_despacho = 100
for cantidad in pedidos:
    if cantidad >= minimo_despacho:
        estado = "DESPACHAR"
    else:

    print(cantidad, "->", estado)

        \item[{\color{arcaRed}\scriptsize\faArrowRight}]

        bateria = 32  # Simulamos recarga completa
