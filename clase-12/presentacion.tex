\sectionframe{¿Todo Número Es Un Número?}{La trampa de las variables numéricas}

\begin{frame}[shrink=5]{La Trampa Numérica}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    \texttt{df.describe()} incluye \texttt{pclass} y \texttt{survived} como numéricas.
    ¿Tiene sentido?}

  \vspace{8pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Operaciones sin sentido:}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item \texttt{pclass.mean()} $= 2.31$

          ¿Qué es la ``clase 2.31''?
        \item \texttt{survived.std()} $= 0.49$

          ¿Desviación estándar de estar vivo?
        \item \texttt{survived.median()} $= 0$

          ¿El pasajero ``mediano'' murió?
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Lo que sí tiene sentido:}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{5pt}
        \item \texttt{pclass.value\_counts()}

          ¿Cuántos en cada clase?
        \item \texttt{survived.value\_counts(normalize=True)}

          ¿Qué proporción sobrevivió?
        \item \texttt{pclass.mode()}

          ¿Cuál es la clase más frecuente?
      \end{itemize}
      }
    \end{column}
  \end{columns}

  \vspace{8pt}
  {\footnotesize\color{arcaDark}
    \faLightbulb\hspace{4pt}
    Que una variable esté almacenada como número \textbf{no significa}
    que sea numérica. \texttt{pclass} y \texttt{survived} son
    \textbf{categóricas disfrazadas}.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Demostración: La Media No Tiene Sentido}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Comparemos operaciones sobre \texttt{pclass} y \texttt{age}.}

  \vspace{4pt}
  \begin{pyblock}
print("=== pclass (categorica disfrazada) ===")
print(f"Media:   {df['pclass'].mean():.2f}")   # 2.31 ???
print(f"Moda:    {df['pclass'].mode()[0]}")     # 3 (tercera clase)
print()
print("=== age (numerica real) ===")
print(f"Media:   {df['age'].mean():.2f}")       # 29.70 tiene sentido
print(f"Mediana: {df['age'].median():.2f}")     # 28.00 tiene sentido
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Para \texttt{age}: media, mediana y std tienen sentido (es una cantidad).
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Para \texttt{pclass}: solo frecuencias y moda tienen sentido (es un grupo).
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=5]{Tipos de Variables}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    Clasificación formal de las variables según su naturaleza.}

  \vspace{6pt}
  {\scriptsize
  \begin{tabular}{@{}llll@{}}
    \toprule
    \textbf{Tipo} & \textbf{Definición} & \textbf{Ejemplo Titanic} & \textbf{Operaciones válidas} \\
    \midrule
    \textbf{Nominal} & Categorías sin orden & sex, embarked, who &
      moda, frecuencia \\
    \textbf{Ordinal} & Categorías con orden & class (1ra > 2da > 3ra) &
      moda, frecuencia, mediana \\
    \textbf{Binaria} & Solo dos valores & survived, alone &
      moda, proporción \\
    \midrule
    \textbf{Discreta} & Números contables & sibsp, parch &
      media, mediana, std \\
    \textbf{Continua} & Números medibles & age, fare &
      media, mediana, std \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\footnotesize\color{arcaDark}
    \faLightbulb\hspace{4pt}
    Las tres primeras son \textbf{categóricas}: representan \textbf{grupos},
    no cantidades. No les calcules media ni desviación estándar.}
\end{frame}

\begin{frame}[shrink=8]{Clasificar las Columnas del Titanic}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    Ejercicio mental: ¿cada columna es categórica o numérica?}

  \vspace{6pt}
  {\scriptsize
  \begin{tabular}{@{}lllp{4cm}@{}}
    \toprule
    \textbf{Columna} & \textbf{dtype} & \textbf{Tipo real} & \textbf{¿Por qué?} \\
    \midrule
    survived & int64 & Binaria & 0/1 = murió/sobrevivió \\
    pclass & int64 & Ordinal & 1, 2, 3 = clase social \\
    sex & object & Nominal & male/female, sin orden \\
    age & float64 & Continua & Años de vida \\
    fare & float64 & Continua & Precio del boleto \\
    embarked & object & Nominal & C/Q/S = puertos \\
    class & category & Ordinal & First/Second/Third \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\footnotesize\color{arcaDark}
    \faExclamationTriangle\hspace{4pt}
    \texttt{pclass} es \texttt{int64} pero NO es numérica.
    \texttt{survived} es \texttt{int64} pero es binaria.
    El \texttt{dtype} \textbf{no determina} el tipo de variable.}
\end{frame}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}

        \item No puedes incluir \texttt{embarked} en un modelo.

\begin{frame}[shrink=21]{Cramér's V: La Matemática}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    La idea: ¿las frecuencias observadas difieren de lo esperado por azar?}

  \vspace{4pt}
  {\scriptsize\bfseries Paso 1: Tabla de contingencia (crosstab)}
  \vspace{2pt}

  {\scriptsize
    Con \texttt{sex} vs \texttt{survived} obtenemos las frecuencias
    \textbf{observadas} ($O_{ij}$). Luego calculamos las
    \textbf{esperadas} ($E_{ij}$): qué esperaríamos si no hubiera
    relación alguna.
  \[
    E_{ij} = \frac{\text{total fila}_i \times \text{total columna}_j}
                   {\text{total general}}
  \]
  }

  \vspace{2pt}
  {\scriptsize\bfseries Paso 2: Chi-cuadrado ($\chi^2$)}
  \vspace{2pt}

  {\scriptsize
    Mide cuánto difieren las frecuencias observadas de las esperadas:
  \[
    \chi^2 = \sum_{i,j}
      \frac{(O_{ij} - E_{ij})^2}{E_{ij}}
  \]
    Si $\chi^2 = 0$: perfecta independencia.
    Mientras más grande, más asociación.
  }

  \vspace{2pt}
  {\scriptsize\bfseries Paso 3: Normalizar $\to$ Cramér's V}
  \vspace{2pt}

  {\scriptsize
  \[
    V = \sqrt{\frac{\chi^2}{n \cdot (k - 1)}}
    \qquad
    \begin{cases}
      n &= \text{número total de observaciones} \\
      k &= \min(\text{filas}, \text{columnas})
    \end{cases}
  \]
    Dividir entre $n \cdot (k-1)$ escala el resultado al rango $[0, 1]$.
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Cramér's V en Código}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Implementación simple con \texttt{scipy.stats}.}

  \vspace{4pt}
  \begin{pyblock}
from scipy.stats import chi2_contingency

def cramers_v(col1, col2):
    tabla = pd.crosstab(col1, col2)
    chi2 = chi2_contingency(tabla)[0]
    n = len(col1)
    k = min(tabla.shape) - 1
    return (chi2 / (n * k)) ** 0.5

v = cramers_v(df["sex"], df["survived"].map(
    {0: "No", 1: "Si"}))
print(f"Cramer's V (sex vs survived): {v:.3f}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      V $\approx$ 0.54 --- asociación fuerte entre sexo y supervivencia.
