\begin{frame}{La Idea: Penalizar los Errores Graves}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    No cambiamos los datos. Le decimos al modelo que
    equivocarse con la minoría \textbf{cuesta más}.}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{textMuted}
        Sin \texttt{class\_weight}:}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item Cada error pesa \textbf{igual}.
        \item Como hay 3x más ``se queda'', el modelo
          prioriza esa clase.
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{arcaRed}
        Con \texttt{class\_weight='balanced'}:}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item Equivocarse con ``renuncia'' pesa
          \textbf{3x más}.
        \item El modelo presta más atención a la minoría.
      \end{itemize}
      }
    \end{column}
  \end{columns}

  \vspace{6pt}
  {\footnotesize\color{arcaDark}
    \faLightbulb\hspace{4pt}
    Volviendo a la certificación: es como si las preguntas
    que no son A \textbf{valieran 3 puntos} cada una.}
\end{frame}

    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      Son \textbf{copias exactas}: el modelo puede memorizarlas.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Oversampling en Código}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Separamos clases, duplicamos la minoría y juntamos
    con \texttt{pd.concat()}.}

  \vspace{4pt}
  \begin{pyblock}
from sklearn.utils import resample

# Separar clases (trabajamos con el DataFrame)
df_may = df_enc[df_enc["Attrition"] == 0]
df_min = df_enc[df_enc["Attrition"] == 1]

# Duplicar minoria hasta igualar
df_min_up = resample(df_min, replace=True,
    n_samples=len(df_may), random_state=42)

# Juntar
df_over = pd.concat([df_may, df_min_up])
print(f"Antes: {len(df_enc)}, Despues: {len(df_over)}")
  \end{pyblock}
\end{frame}

\begin{frame}{SMOTE: Crear Datos Nuevos}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    SMOTE no copia: crea \textbf{ejemplos nuevos} interpolando
    entre vecinos de la clase minoritaria.}

  \vspace{6pt}

  \begin{tikzpicture}[
    dot/.style={circle, minimum size=0.3cm, inner sep=0pt},
    new/.style={circle, minimum size=0.3cm, inner sep=0pt,
                draw=codeGreen, thick, dashed},
  ]
    \node[dot, fill=arcaRed] (a) at (0, 0) {};
    \node[dot, fill=arcaRed] (b) at (2.5, 1.5) {};
    \node[dot, fill=arcaRed] (c) at (2, -0.6) {};
    \draw[dashed, arcaRed!40, thick] (a) -- (b);
    \draw[dashed, arcaRed!40, thick] (a) -- (c);
    \draw[dashed, arcaRed!40, thick] (b) -- (c);
    \node[new, fill=codeGreen!30] at (1, 0.6) {};

\begin{frame}[fragile,shrink=28]{Los 4 Modelos}

  \begin{pyblock}
modelos = {
  "Base":       (X_s, y, None),
  "Balanced":   (X_s, y, "balanced"),
  "Oversample": (X_over, y_over, None),
  "SMOTE":      (X_smote, y_smote, None),
}
preds = {}
for nombre, (X_tr, y_tr, cw) in modelos.items():
    m = LogisticRegression(max_iter=1000,
                           class_weight=cw)
    m.fit(X_tr, y_tr)
    preds[nombre] = m.predict(X_s)
  \end{pyblock}
\end{frame}

\begin{frame}[fragile,shrink=28]{Tabla de Resultados}

print(f"{'Metodo':<12} {'Precision':>10} {'Recall':>10}"
      f" {'VP':>5} {'FN':>5}")
print("-" * 48)
for nombre, pred in preds.items():
    cm_r = confusion_matrix(y, pred)
    prec = precision_score(y, pred)
    rec  = recall_score(y, pred)
    vp, fn = cm_r[1, 1], cm_r[1, 0]
    print(f"{nombre:<12} {prec:>10.3f} {rec:>10.3f}"
          f" {vp:>5d} {fn:>5d}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}

      VP = empleados que renuncian y detectamos.

  }

\begin{frame}{¿Cuál Método Elijo?}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    No hay un ganador universal. Depende del costo del error.}

  \vspace{8pt}
  {\scriptsize
  \begin{tabular}{@{}p{2.5cm}p{4cm}p{4cm}@{}}
    \toprule
    \textbf{Método} & \textbf{Cuándo usarlo} & \textbf{Limitación} \\
    \midrule
    \texttt{class\_weight} &
      Siempre es un buen primer intento.
      No modifica los datos. &
      A veces no es suficiente. \\[6pt]
    Oversampling &
      Datasets pequeños.
      No quieres perder datos. &
      Copias exactas: riesgo de
      que el modelo memorice. \\[6pt]
    SMOTE &
      Generalmente mejor que copiar.
      Crea datos diferentes. &
      Necesita instalar
      \texttt{imbalanced-learn}. \\
    \bottomrule
  \end{tabular}
  }

\sectionframe{Train / Test Split}{Evaluar de verdad}

\begin{frame}{¿Por Qué No Podemos Evaluar con los Mismos Datos?}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    Imaginen que un profesor les da el examen final para que
    ``practiquen''\ldots\ y luego les toma \textbf{exactamente el
    mismo examen}.}

  \vspace{6pt}
  {\footnotesize
  \begin{itemize}\setlength{\itemsep}{6pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Todos sacan 10. Pero ¿realmente aprendieron?
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Quizás solo \textbf{memorizaron las respuestas}.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Si les cambian los números o el orden de las preguntas,
      algunos fallan.
  \end{itemize}
  }

  \vspace{6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Eso es lo que hicimos hoy: entrenar y evaluar con los mismos datos.
    El modelo puede estar \textbf{memorizando} en vez de aprendiendo.}
\end{frame}

\begin{frame}{La Solución: Separar los Datos}
  \vspace{-4pt}
  {\footnotesize\color{arcaDark}\bfseries
    Dividimos los datos en dos partes \textbf{antes} de entrenar.}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, minimum width=3.5cm,
                minimum height=1cm, draw, font=\footnotesize\bfseries,
                align=center, inner sep=4pt},
  ]
    \node[box, fill=arcaRed!15, draw=arcaRed] (all) at (0, 0.5) {Todos los datos};

    % Train
    \node[box, fill=codeBlue!15, draw=codeBlue, minimum width=5cm] (train)
      at (-2.5, -1.5) {Train (80\%)};
    \node[font=\scriptsize\color{textMuted}, below=2pt of train]
      {Practicar con ejercicios};

    % Test
    \node[box, fill=codeGreen!15, draw=codeGreen, minimum width=3cm] (test)
      at (3.5, -1.5) {Test (20\%)};
    \node[font=\scriptsize\color{textMuted}, below=2pt of test]
      {El examen final};

    \draw[->, thick, arcaRed!60] (all.south west) -- (train.north);
    \draw[->, thick, arcaRed!60] (all.south east) -- (test.north);

    % Muro
    \draw[ultra thick, arcaDark!40, dashed] (1, -0.5) -- (1, -2.5);
    \node[font=\tiny\color{arcaDark}\bfseries, rotate=90] at (1.3, -1.5) {MURO};
  \end{tikzpicture}
  \end{center}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      El modelo \textbf{aprende} con train (como estudiar).
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Lo \textbf{evaluamos} con test (como el examen).
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      El modelo \textbf{nunca} ve el test durante el entrenamiento.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Train/Test Split con Iris}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Usamos \texttt{train\_test\_split} de sklearn.}

  \vspace{4pt}
  \begin{pyblock}
from sklearn.model_selection import train_test_split

iris = sns.load_dataset("iris")
iris["es_setosa"] = (iris["species"] == "setosa").astype(int)

X_iris = iris[["sepal_length", "sepal_width",
               "petal_length", "petal_width"]]
y_iris = iris["es_setosa"]

X_train, X_test, y_train, y_test = train_test_split(
    X_iris, y_iris,
    test_size=0.2,       # 20% para test
    random_state=42)

print(f"Train: {len(X_train)}, Test: {len(X_test)}")
  \end{pyblock}
\end{frame}

\begin{frame}[fragile,shrink=28]{Entrenar con Train, Evaluar con Test}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    El modelo aprende con train y lo evaluamos con datos que
    \textbf{nunca vio}.}

  \vspace{4pt}

modelo_iris = LogisticRegression(max_iter=1000)
modelo_iris.fit(X_train, y_train)      # estudiar
y_pred_iris = modelo_iris.predict(X_test)  # examen

print(f"Accuracy en test: "
      f"{accuracy_score(y_test, y_pred_iris):.3f}")
print(f"Recall en test:   "
      f"{recall_score(y_test, y_pred_iris):.3f}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{codeGreen}\scriptsize\faCheckCircle}]
      Esta evaluación es \textbf{justa}: el modelo nunca
      vio los datos de test.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      La próxima clase aplicamos esto a nuestros modelos
      de desbalance, con curva ROC y umbral óptimo.
  \end{itemize}
  }
