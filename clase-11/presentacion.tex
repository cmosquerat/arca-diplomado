\begin{frame}[shrink=5]{¿Cómo Comparar Dos Variables?}
  \vspace{-4pt}
  {\footnotesize
    Hasta ahora analizamos variables \textbf{una a la vez} (histograma, boxplot, estadísticas).
    Ahora queremos responder: \textbf{¿hay relación entre dos variables?}}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=5.5cm, minimum height=1.6cm,
                font=\scriptsize\color{textDark}, align=left, text width=5cm,
                inner sep=6pt},
  ]
    \node[box=codeBlue] (q) at (0,0)
      {\textbf{\color{codeBlue}La pregunta}\\[2pt]
       ``Los vinos con más alcohol,
       ¿tienden a tener mejor nota?''\\
       ``¿Más acidez significa peor vino?''};
    \node[box=codeGreen] (a) at (6.5,0)
      {\textbf{\color{codeGreen}La herramienta}\\[2pt]
       El \textbf{scatter plot} (gráfico de dispersión):
       cada punto es un vino, un eje por variable.
       Buscamos \textbf{patrones} en la nube de puntos.};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ El scatter plot es el primer paso.
    Después le pondremos un \textbf{número} a esa relación (la correlación).}
\end{frame}

\begin{frame}[shrink=5]{Anatomía de un Scatter Plot}
  \vspace{-4pt}
  {\footnotesize
    Cada \textbf{punto} en el gráfico representa una observación (en nuestro caso, un vino):}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    step/.style={rectangle, rounded corners=3pt, draw=#1, fill=#1!8,
                 minimum width=12cm, minimum height=0.7cm,
                 font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                 inner sep=5pt},
  ]
    \node[step=codeBlue] (s1) at (0, 0)
      {\textbf{Eje X:} una variable (ej: \texttt{alcohol}).
       La posición horizontal del punto es su valor de alcohol.};
    \node[step=codePurple, below=0.15cm of s1] (s2)
      {\textbf{Eje Y:} otra variable (ej: \texttt{quality}).
       La posición vertical del punto es su nota de calidad.};
    \node[step=codeGreen, below=0.15cm of s2] (s3)
      {\textbf{Un punto = un vino.} Si tiene alcohol $= 12$ y quality $= 7$,
       el punto está en la posición $(12, 7)$.};
    \node[step=arcaRed, below=0.15cm of s3] (s4)
      {\textbf{El patrón:} si los puntos ``suben'' de izquierda a derecha $\to$
       relación positiva. Si ``bajan'' $\to$ negativa. Si no hay patrón $\to$ sin relación.};
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ Con 1599 vinos, tendremos 1599 puntos.
    Usamos \texttt{alpha} (transparencia) para ver dónde se acumulan.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Nuestro Primer Scatter Plot}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    ¿Los vinos con más alcohol tienen mejor nota?}

  \vspace{4pt}
  \begin{pyblock}
sns.scatterplot(x="alcohol", y="quality",
                data=df, alpha=0.3)
plt.xlabel("Alcohol (% vol)")
plt.ylabel("Quality (nota 0-10)")
plt.title("Alcohol vs Quality")
plt.show()
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{x=} y \texttt{y=}: las dos variables a comparar.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{alpha=0.3}: transparencia (0 = invisible, 1 = sólido).
      Con muchos puntos encimados, ayuda a ver dónde se concentran.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Scatter Plot: Relación Negativa}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries

  \vspace{4pt}

sns.scatterplot(x="volatile acidity", y="quality",

\end{frame}

\begin{frame}[shrink=5]{Del Scatter Plot a un Número: La Correlación}
  \vspace{-4pt}
  {\footnotesize
    El scatter plot nos muestra el patrón \textbf{visualmente}.
    Pero queremos un \textbf{número} que resuma esa relación:}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=3.5cm, minimum height=1.8cm,
                font=\scriptsize\color{textDark}, align=center, text width=3.2cm},
  ]
    \node[box=codeGreen] (pos) at (0,0)
      {\textbf{\color{codeGreen}Correlación positiva}\\[2pt]
       $r \approx +1$\\
       Puntos ``suben''\\
       {\tiny alcohol $\uparrow$ quality $\uparrow$}};
    \node[box=textMuted] (zero) at (4.5,0)
      {\textbf{\color{textMuted}Sin correlación}\\[2pt]
       $r \approx 0$\\
       Puntos sin patrón\\
       {\tiny azúcar vs quality}};
    \node[box=arcaRed] (neg) at (9,0)
      {\textbf{\color{arcaRed}Correlación negativa}\\[2pt]
       $r \approx -1$\\
       Puntos ``bajan''\\
       {\tiny acidez $\uparrow$ quality $\downarrow$}};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      La \textbf{correlación} es ese número: va de $-1$ a $+1$.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Más cerca de $\pm 1$ = relación más fuerte. Cerca de $0$ = relación débil.
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      \textbf{$r \approx 0$ no significa ``no hay relación''} --- solo que no hay relación \textbf{lineal}.
  \end{itemize}
  }
\end{frame}

\sectionframe{Pearson y Spearman}{Dos formas de medir la relación}

\begin{frame}[shrink=10]{Pearson ($r$): De Dónde Sale la Fórmula}
  \vspace{-4pt}
  {\footnotesize
    Queremos un número que diga: ``cuando $x$ sube, ¿$y$ también sube?''
    Construyamos la idea paso a paso:}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    step/.style={rectangle, rounded corners=3pt, draw=#1, fill=#1!8,
                 minimum width=12cm, minimum height=0.8cm,
                 font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                 inner sep=5pt},
  ]
    \node[step=codeBlue] (s1) at (0, 0)
      {\textbf{1. Desviación:} $(x_i - \bar{x})$ mide cuánto se aleja $x_i$
       de su media. Igual para $y$: $(y_i - \bar{y})$.};
    \node[step=codePurple, below=0.15cm of s1] (s2)
      {\textbf{2. Producto:} $(x_i - \bar{x})(y_i - \bar{y})$.
       Ambos arriba de su media $\to$ positivo.
       Uno arriba y otro abajo $\to$ negativo.};
    \node[step=codeGreen, below=0.15cm of s2] (s3)
      {\textbf{3. Sumar:} $\sum(x_i - \bar{x})(y_i - \bar{y})$.
       Mayoría positivos $\to$ relación positiva. Negativos $\to$ negativa.};
    \node[step=arcaRed, below=0.15cm of s3] (s4)
      {\textbf{4. Normalizar:} Dividimos entre $(n-1)\, s_x \, s_y$ para que $r$
       quede entre $-1$ y $+1$.};
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}
    \node[rectangle, rounded corners=4pt, draw=arcaRed, fill=arcaRed!6,
          minimum width=6cm, minimum height=1.1cm,
          font=\large\color{arcaDark}, align=center] {
      $r = \dfrac{\sum_{i=1}^{n}(x_i - \bar{x})(y_i - \bar{y})}{(n-1)\, s_x \, s_y}$
    };
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}[shrink=15]{Spearman ($\rho$): Pearson Pero con Rangos}
  \vspace{-4pt}
  {\footnotesize
    Spearman tiene una idea elegante: en vez de usar los \textbf{valores} originales,
    usa sus \textbf{posiciones} (rangos) cuando los ordenas de menor a mayor.}

  \vspace{4pt}
  \begin{center}
  {\small
  \renewcommand{\arraystretch}{1.2}
  \begin{tabular}{@{}cccc@{}}
    \toprule
    \textbf{Dato $x$} & \textbf{Rango de $x$} & \textbf{Dato $y$} & \textbf{Rango de $y$} \\
    \midrule
    3.2 & 1 & 5 & 1 \\
    4.1 & 2 & 6 & 2 \\
    8.7 & 3 & 7 & 3 \\
    150.0 {\tiny(outlier!)} & 4 & 8 & 4 \\
    \bottomrule
  \end{tabular}
  }
  \end{center}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=5.5cm, minimum height=1.5cm,
                font=\scriptsize\color{textDark}, align=left, text width=5cm,
                inner sep=6pt},
  ]
    \node[box=codePurple] (how) at (0,0)
      {\textbf{\color{codePurple}¿Cómo se calcula?}\\[2pt]
       Reemplaza cada valor por su rango
       y aplica la \textbf{fórmula de Pearson} sobre esos rangos.};
    \node[box=codeGreen] (why) at (6.5,0)
      {\textbf{\color{codeGreen}¿Por qué funciona?}\\[2pt]
       El outlier 150.0 solo tiene rango 4.
       No ``arrastra'' el resultado.
       Mide si el \textbf{orden} se mantiene.};
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}[shrink=5]{Pearson vs Spearman: Resumen}
  \vspace{-4pt}
  {\small
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}p{3cm}p{4.5cm}p{4.5cm}@{}}
    \toprule
    & \textbf{Pearson ($r$)} & \textbf{Spearman ($\rho$)} \\
    \midrule
    \textbf{Mide} &
      Relación \textbf{lineal} &
      Relación \textbf{monótona} \\
    \textbf{Usa} &
      Los \textbf{valores} originales &
      Los \textbf{rangos} (posiciones) \\
    \textbf{Outliers} &
      Sensible (un extremo lo distorsiona) &
      Robusto (el rango no cambia mucho) \\
    \textbf{Normalidad} &
      Funciona mejor con datos normales &
      No la requiere \\
    \textbf{Cuándo usar} &
      Datos normales, sin outliers, relación recta &
      Datos sesgados, outliers, relación curva \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ Si Pearson y Spearman dan resultados muy diferentes,
    probablemente hay \textbf{outliers} o una relación \textbf{no lineal}.}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ \textbf{¿Qué es ``monótona''?} Una relación donde, al subir $x$, $y$ \textbf{siempre sube}
    (o siempre baja), aunque no sea en línea recta. Ejemplo: una curva logarítmica sube
    siempre, pero cada vez más lento. Pearson no la detecta bien; Spearman sí.}
\end{frame}

\begin{frame}[shrink=8]{¿Por Qué Pearson Puede Fallar?}
  \vspace{-4pt}
  {\footnotesize
    Pearson mide relación \textbf{lineal}. Si la relación es curva,
    Pearson dirá ``no hay relación'' cuando \textbf{sí la hay}:}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=3.8cm, minimum height=1.8cm,
                font=\scriptsize\color{textDark}, align=center, text width=3.5cm},
  ]
    \node[box=codeGreen] (lin) at (0,0)
      {\textbf{\color{codeGreen}Lineal}\\[2pt]
       $y = a + bx$\\
       Pearson funciona bien\\
       $r = 0.95$};
    \node[box=codeOrange] (quad) at (4.8,0)
      {\textbf{\color{codeOrange}Cuadrática}\\[2pt]
       $y = a + bx + cx^2$\\
       Pearson dice $r \approx 0$\\
       ¡Pero la relación existe!};
    \node[box=codePurple] (log) at (9.6,0)
      {\textbf{\color{codePurple}Logarítmica}\\[2pt]
       $y = a + b \ln(x)$\\
       Pearson subestima\\
       Spearman captura mejor};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faExclamationTriangle}]
      \textbf{Moraleja:} Siempre haz el scatter plot \textbf{antes} de confiar en el número $r$.
    \item[{\color{codeBlue}\scriptsize\faArrowRight}]
      Spearman detecta relaciones monótonas (siempre sube o siempre baja),
      aunque no sean lineales.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=10]{¿Qué Hace \texttt{regplot}? --- La Recta de Mejor Ajuste}
  \vspace{-4pt}
  {\footnotesize
    \texttt{sns.regplot()} dibuja los puntos \textbf{y además ajusta una línea}.
    Tiene dos modos:}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codeBlue} Modo lineal (por defecto)}
      \vspace{4pt}

      {\scriptsize
      Busca la recta $y = a + bx$ que \textbf{minimiza}
      la suma de errores al cuadrado:
      }

      \vspace{4pt}
      \begin{center}
      \begin{tikzpicture}
        \node[rectangle, rounded corners=3pt, draw=codeBlue, fill=codeBlue!6,
              font=\small\color{arcaDark}, inner sep=6pt] {
          $\min \displaystyle\sum_{i=1}^{n} (y_i - a - bx_i)^2$
        };
      \end{tikzpicture}
      \end{center}

      \vspace{2pt}
      {\scriptsize
      Cada $(y_i - a - bx_i)$ es la distancia del punto a la recta.
      La recta con \textbf{menos error} es la elegida.
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codePurple} Modo LOWESS (\texttt{lowess=True})}
      \vspace{4pt}

      {\scriptsize
      En vez de \textbf{una} recta para todos, ajusta
      \textbf{muchas rectas locales}:

      \begin{itemize}\setlength{\itemsep}{3pt}
        \item Para cada punto, toma los vecinos cercanos.
        \item Ajusta una recta solo con esos vecinos.
        \item Resultado: una \textbf{curva suave}.
      \end{itemize}

      Útil cuando la relación \textbf{no es una recta}.
      }
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ El modo lineal es el \textbf{método de mínimos cuadrados}.}
\end{frame}

    \faLightbulb\ \texttt{lowess=True} ajusta una curva suave en vez de una recta.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Los Outliers de la Clase Pasada: ¿Buenos o Malos?}

  {\footnotesize\color{arcaDark}\bfseries
    En la clase pasada encontramos outliers en chlorides. ¿Qué calidad tienen?}

  \vspace{4pt}
  \begin{pyblock}
col = "chlorides"
Q1 = df[col].quantile(0.25)
Q3 = df[col].quantile(0.75)
IQR = Q3 - Q1

outliers = df[(df[col] < Q1 - 1.5*IQR) |
              (df[col] > Q3 + 1.5*IQR)]
normales = df[~df.index.isin(outliers.index)]

print(f"Outliers: {len(outliers)} vinos")
print(f"  Quality promedio: {outliers['quality'].mean():.2f}")
print(f"Normales: {len(normales)} vinos")
print(f"  Quality promedio: {normales['quality'].mean():.2f}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ Si los outliers tienen quality más baja,
    la sal excesiva \textbf{arruina} el vino --- ¡eso es información útil!}

\begin{frame}[fragile,shrink=28]{¿Afectan los Outliers a la Correlación?}

  \begin{pyblock}
col = "chlorides"
Q1 = df[col].quantile(0.25)
Q3 = df[col].quantile(0.75)

df_clean = df[(df[col] >= Q1 - 1.5*IQR) &
              (df[col] <= Q3 + 1.5*IQR)]

r_con = df[col].corr(df["quality"])
r_sin = df_clean[col].corr(df_clean["quality"])
rho_con = df[col].corr(df["quality"],
                        method="spearman")
rho_sin = df_clean[col].corr(df_clean["quality"],
                              method="spearman")
print(f"Con outliers:  Pearson={r_con:.3f}, "
      f"Spearman={rho_con:.3f}")
print(f"Sin outliers:  Pearson={r_sin:.3f}, "
      f"Spearman={rho_sin:.3f}")
