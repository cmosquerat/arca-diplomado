\sectionframe{El Problema}{¿Qué pasa cuando el código se repite?}

% ─────────────────────────────────────────────────────────────────────────────
%  CÓDIGO REPETIDO
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{El Código se Repite}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Tres clientes, el mismo cálculo copiado tres veces:}

  \vspace{2pt}
  \begin{pyblock}
cajas_1, precio_1 = 30, 12.50
ingreso_1 = cajas_1 * precio_1
print(f"Cliente 1: {ingreso_1}")

cajas_2, precio_2 = 15, 12.50
ingreso_2 = cajas_2 * precio_2
print(f"Cliente 2: {ingreso_2}")

cajas_3, precio_3 = 50, 12.50
ingreso_3 = cajas_3 * precio_3
print(f"Cliente 3: {ingreso_3}")
  \end{pyblock}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      Si el jefe dice ``ahora el precio incluye IVA'',
      hay que cambiar \textbf{3 lugares}.
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      Con 50 clientes serían 50 bloques idénticos.
    \item[{\color{arcaRed}\scriptsize\faBolt}]
      \textbf{¿Hay una forma de escribir la fórmula una sola vez?}
  \end{itemize}
  }
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  LA SOLUCIÓN: FUNCIONES
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{La Solución: Funciones}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      {\footnotesize\color{arcaRed}\bfseries\faTimes\hspace{4pt} Antes (repetido):}
      \begin{pyblock}
cajas_1, precio_1 = 30, 12.50
ingreso_1 = cajas_1 * precio_1

cajas_2, precio_2 = 15, 12.50
ingreso_2 = cajas_2 * precio_2
      \end{pyblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      {\footnotesize\color{codeGreen}\bfseries\faCheck\hspace{4pt} Después (función):}
      \begin{pyblock}
def calcular_ingreso(cajas, precio):
    return cajas * precio

ingreso_1 = calcular_ingreso(30, 12.50)
ingreso_2 = calcular_ingreso(15, 12.50)
      \end{pyblock}
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      La fórmula se escribe \textbf{una sola vez} dentro de la función.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Si cambia el cálculo, se modifica \textbf{un solo lugar}.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Se puede usar con cualquier cliente, las veces que haga falta.
  \end{itemize}
  }
\end{frame}

def calcular_ingreso(cajas, precio):

      \vspace{4pt}
      \begin{center}
      \begin{tikzpicture}[
        node distance=0.25cm,

        \node[lbl, below=of d] (n) {\texttt{calcular\_ingreso}};
        \node[desc, right=0.15cm of n] (n2) {nombre de la función};

        \node[lbl, below=of n] (p) {\texttt{(cajas, precio)}};
        \node[desc, right=0.15cm of p] (p2) {parámetros (entradas)};

        \node[lbl, below=of p] (c) {bloque indentado};
        \node[desc, right=0.15cm of c] (c2) {cuerpo de la función};

        \node[lbl, below=of c] (r) {\texttt{return}};
        \node[desc, right=0.15cm of r] (r2) {salida (opcional)};
      \end{tikzpicture}
      \end{center}
    \end{column}

    \begin{column}{0.46\textwidth}
      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{def} inicia la definición.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          El nombre describe \textbf{qué hace}
          (verbo + sustantivo).
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Los parámetros son las entradas que
          la función necesita para trabajar.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          El cuerpo contiene la lógica,
          \textbf{indentado} con 4 espacios.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{return} devuelve el resultado.
          Sin él, la función devuelve \texttt{None}.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=28]{Anatomía de una Función --- Cuerpo y Llamada}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.42\textwidth}
      \vspace{-2pt}
      \begin{center}
      \begin{tikzpicture}[

def calcular_ingreso(cajas, precio):

print("Ingreso:", resultado)
      \end{pyblock}

      \begin{outputblock}

      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{return} devuelve el resultado.
          Sin él, la función devuelve \texttt{None}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Al llamar la función, \texttt{30} entra en
          \texttt{cajas} y \texttt{12.50} entra en \texttt{precio}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]

      }

  \end{columns}

% ─────────────────────────────────────────────────────────────────────────────
%  PARÁMETROS: QUÉ SON Y CÓMO FLUYEN
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{Los Parámetros Son la Pieza Clave}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Sin parámetros, una función siempre haría lo mismo.
    Con ellos, es reutilizable.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Sin parámetros --- siempre hace lo mismo}
      \begin{pyblock}
def ingreso_fijo():
    return 30 * 12.50

print(ingreso_fijo())
print(ingreso_fijo())
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
375.0
375.0
      \end{outputblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Con parámetros --- flexible}
      \begin{pyblock}
def ingreso(cajas, precio):
    return cajas * precio

print(ingreso(30, 12.50))
print(ingreso(15, 9.00))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
375.0
135.0
      \end{outputblock}
    \end{column}
  \end{columns}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Los parámetros son \textbf{variables temporales}
      que solo existen dentro de la función.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Cada vez que llaman la función, esas variables
      reciben valores nuevos.
  \end{itemize}
  }
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  PARÁMETROS POSICIONALES VS POR NOMBRE
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{El Orden Importa --- Parámetros Posicionales}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Python asigna los valores en el orden en que los pasan:}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{pyblock}
def describir(producto, cantidad, precio):
    total = cantidad * precio
    return f"{producto}: {cantidad} x {precio} = {total}"

print(describir("Coca-Cola", 30, 12.50))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Coca-Cola: 30 x 12.5 = 375.0
      \end{outputblock}

      \vspace{2pt}
      {\scriptsize\color{codeOrange}\bfseries\faExclamationTriangle\hspace{2pt}
        Si intercambian el orden:}
      \begin{pyblock}
print(describir(30, "Coca-Cola", 12.50))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
ERROR: no se puede multiplicar str x float
      \end{outputblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{2pt}
        Pueden usar el nombre para evitar errores:}
      \begin{pyblock}
print(describir(
    precio=12.50,
    cantidad=30,
    producto="Coca-Cola"
))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Coca-Cola: 30 x 12.5 = 375.0
      \end{outputblock}

      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Si pasan los argumentos \textbf{por nombre},
          el orden no importa.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Esto hace el código más legible
          cuando hay muchos parámetros.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}[fragile,shrink=28]{¿Qué Pasa si Paso Más o Menos Argumentos?}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Python es estricto: la cantidad de argumentos debe coincidir.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Faltan argumentos:}

    \item[{\color{arcaRed}\scriptsize\faArrowRight}]

  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}

equipo = [

% ─────────────────────────────────────────────────────────────────────────────
%  ¿POR QUÉ VALIDAR?

\begin{frame}[fragile,shrink=28]{¿Qué Pasa Cuando los Datos Son Incorrectos?}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Sin validación, la función puede dar resultados absurdos o explotar.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Sin protección --- resultados absurdos}
      \begin{pyblock}
def ingreso(cajas, precio):
    return cajas * precio

print(ingreso(-5, 12.50))
print(ingreso(30, 0))
print(ingreso(30, -10))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
-62.5
0
-300
      \end{outputblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Con validación --- se detecta el problema}
      \begin{pyblock}
def ingreso(cajas, precio):
    if cajas < 0:
        print("Error: cajas negativas")
        return None
    if precio <= 0:
        print("Error: precio invalido")
        return None
    return cajas * precio

print(ingreso(-5, 12.50))
print(ingreso(30, 12.50))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Error: cajas negativas
None
375.0
      \end{outputblock}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  PATRÓN: GUARDIA
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{Patrón Guardia --- Validar Antes de Calcular}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Las validaciones van \textbf{al inicio}:
    si algo está mal, la función se detiene temprano.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
def procesar_entrega(cliente, pedido,
                     carga, precio):
    if pedido <= 0:
        return {"error": "pedido invalido"}
    if pedido > carga:
        return {"error": "sin stock"}
    if precio <= 0:
        return {"error": "precio invalido"}

    ingreso = pedido * precio
    return {
        "cliente": cliente,
        "ingreso": ingreso,
        "restante": carga - pedido
    }
      \end{pyblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \begin{pyblock}
r1 = procesar_entrega("Sol", 30, 100, 12.5)
print(r1)
r2 = procesar_entrega("Sol", -5, 100, 12.5)
print(r2)
r3 = procesar_entrega("Sol", 30, 10, 12.5)
print(r3)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
resultado: {'cliente':'Sol','ingreso':375.0,
  'restante':70}
resultado: {'error':'pedido invalido'}
resultado: {'error':'sin stock'}
      \end{outputblock}

      \vspace{2pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Cada \texttt{return} temprano
          actúa como un \textbf{guardia}.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          El cálculo solo se ejecuta
          si todos los guardias pasaron.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  EJERCICIO: AGREGAR VALIDACIÓN
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=28]{Práctica: Blindar una Función \hfill
    {\normalsize\color{arcaRed}\faClock\ 7 min}}
  \vspace{-6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Agreguen validaciones a esta función para que nunca
    dé resultados absurdos:}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
def ficha_venta(producto, cantidad, precio):
    subtotal = cantidad * precio
    iva = subtotal * 0.12
    total = subtotal + iva
    return {
        "producto": producto,
        "total": total
    }
      \end{pyblock}

      \vspace{2pt}
      {\footnotesize\color{arcaDark}\bfseries Validaciones requeridas:}
      {\footnotesize
      \begin{enumerate}\setlength{\itemsep}{2pt}
        \item \texttt{cantidad} debe ser $> 0$
        \item \texttt{precio} debe ser $> 0$
        \item \texttt{producto} no puede estar vacío
              (\texttt{""})
      \end{enumerate}
      }
    \end{column}

    \begin{column}{0.44\textwidth}
      {\scriptsize\color{textMuted}\bfseries Pruébenla con:}
      \begin{pyblock}
print(ficha_venta("Coca", 30, 12.50))
print(ficha_venta("", 30, 12.50))
print(ficha_venta("Coca", -1, 12.50))
print(ficha_venta("Coca", 30, 0))
      \end{pyblock}

      \vspace{4pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Si algo está mal, retornar un dict
          con clave \mintinline{python}{"error"}
          y un mensaje descriptivo.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Primero las validaciones,
          después el cálculo.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}
