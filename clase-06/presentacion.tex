\sectionframe{Los Tres Tipos de Error}{¿Por qué mi código no funciona?}

\begin{frame}[fragile,shrink=28]{Los Tres Tipos de Error}
  \vspace{-6pt}
  {\footnotesize
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Tipo} & \textbf{Cuándo} & \textbf{Ejemplo} \\
    \midrule
    Sintaxis & No arranca & Falta \texttt{:} \\
    Runtime  & Al ejecutar & División por cero \\
    Lógico   & Da resultado mal & \texttt{+} en vez de \texttt{*} \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries Sintaxis:}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
def ingreso(cajas, precio)
    return cajas * precio
      \end{pyblock}

      \vspace{2pt}
      {\scriptsize\color{codeOrange}\bfseries Runtime:}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
def promedio(lista):
    return sum(lista) / len(lista)
promedio([])  # ZeroDivisionError
      \end{pyblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeBlue}\bfseries Lógico:}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
def ingreso(cajas, precio):
    return cajas + precio  # deberia ser *
print(ingreso(30, 12.50))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}[minted options={fontsize=\scriptsize, breaklines}]
42.5
      \end{outputblock}

      \vspace{2pt}
      {\scriptsize\color{textMuted}
        \faExclamationTriangle\ El lógico es el más difícil:
        no da error, pero el resultado está \textbf{mal}.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=28]{Quiz: ¿Qué Tipo de Error Es?}
  \vspace{-6pt}

  \vspace{8pt}
  {\scriptsize\color{textMuted}

    \item[{\color{arcaRed}\scriptsize\faArrowRight}]

\sectionframe{\texttt{try/except}}{Código que no explota}

\begin{frame}[fragile,shrink=28]{\texttt{try/except} --- La Red de Seguridad}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Sin protección --- explota}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
def promedio(lista):
    return sum(lista) / len(lista)

print(promedio([]))
print("Nunca llega aqui")
      \end{pyblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Con \texttt{try/except} --- sigue}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
def promedio(lista):
    try:
        return sum(lista) / len(lista)
    except ZeroDivisionError:
        return 0

print(promedio([]))
print("El programa sigue")
      \end{pyblock}
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{try}: ``intenta ejecutar este código''.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{except ZeroDivisionError}: ``si falla con ESE error, haz esto''.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      El programa \textbf{no se detiene}.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Flujo Completo: \texttt{try/except/else/finally}}
  \vspace{-6pt}
  \begin{pyblock}
def leer_precio(texto):
    try:
        precio = float(texto)
    except ValueError:
        print(f"'{texto}' no es numero")
        return None
    else:
        print(f"Precio valido: {precio}")
        return precio
    finally:
        print("--- Proceso terminado ---")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{else}: se ejecuta si \textbf{no} hubo error.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{finally}: se ejecuta \textbf{siempre}, haya error o no.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{\texttt{except} Específico vs Genérico}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Genérico --- oculta bugs}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
try:
    resultado = 100 / 0
except:
    print("Algo fallo")
    resultado = 0
      \end{pyblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Específico --- sabes qué pasó}
      \begin{pyblock}[minted options={fontsize=\scriptsize, breaklines}]
try:
    resultado = 100 / 0
except ZeroDivisionError:
    print("Division por cero")
    resultado = 0
      \end{pyblock}
    \end{column}
  \end{columns}

  \vspace{6pt}
  {\footnotesize\color{arcaDark}\bfseries
    Regla: cada \texttt{except} debe nombrar el error que espera.}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
    Si no sabes cuál puede ocurrir, deja que explote ---
    así aprendes qué es.}
\end{frame}

\begin{frame}[fragile,shrink=28]{\texttt{try/except} en Funciones Reales}
  \vspace{-6pt}
  \begin{pyblock}
def procesar_venta(vendedor):
    try:
        nombre = vendedor["nombre"]
        ventas = vendedor["ventas"]
        meta = vendedor["meta"]
    except KeyError as e:
        return {"error": f"Falta el campo {e}"}
    if ventas < 0:
        return {"error": "Ventas negativas"}
    bono = ventas * 0.10 if ventas >= meta else 0
    return {"nombre": nombre, "bono": bono}
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{try/except} para errores de \textbf{datos}.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{if} para reglas de \textbf{negocio}.
  \end{itemize}
  }
\end{frame}

  \end{itemize}

\begin{frame}[fragile,shrink=28]{¿Qué Es un Archivo CSV?}
  \vspace{-6pt}
  {\footnotesize

\begin{frame}[fragile,shrink=28]{Seleccionar y Filtrar Datos}
  \vspace{-6pt}
  \begin{pyblock}
import pandas as pd
df = pd.read_csv("datos_ventas.csv")

# Seleccionar una columna
print(df["nombre"])

# Crear columna nueva
df["cumple_meta"] = df["ventas"] >= df["meta"]

# Filtrar filas
sobre_meta = df[df["ventas"] >= df["meta"]]
print(f"{len(sobre_meta)} vendedores sobre meta")
print(sobre_meta[["nombre", "ventas", "meta"]])
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{df["col"]} $\to$ una columna.
      \texttt{df[["a","b"]]} $\to$ varias.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \texttt{df[condición]} $\to$ filtra filas.
  \end{itemize}
  }
\end{frame}

\begin{frame}[fragile,shrink=28]{Agrupar y Resumir: \texttt{groupby()}}
  \vspace{-6pt}
  {\footnotesize
    Equivalente a una tabla dinámica de Excel.}

  \vspace{4pt}
  \begin{pyblock}
import pandas as pd
df = pd.read_csv("datos_ventas.csv")

por_zona = df.groupby("zona")["ventas"].agg(
    ["sum", "mean", "count"]
)
print(por_zona)

print(f"\nTotal: ${df['ventas'].sum():,}")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize
  \begin{tabular}{@{}ll@{}}
    \texttt{.sum()} & Suma total \\
    \texttt{.mean()} & Promedio \\
    \texttt{.count()} & Cantidad \\
  \end{tabular}
  }
\end{frame}

\begin{frame}[fragile,shrink=31]{Exportar: \texttt{df.to\_csv()}}
  \vspace{-6pt}
  \begin{pyblock}
import pandas as pd
df = pd.read_csv("datos_ventas.csv")

df["comision"] = df.apply(
    lambda v: v["ventas"] * 0.05
              if v["ventas"] >= v["meta"] else 0,
    axis=1
)

df["estado"] = df.apply(
    lambda v: "Estrella" if v["ventas"] >= 2*v["meta"]
              else "Cumple" if v["ventas"] >= v["meta"]
              else "Bajo",
    axis=1
)

df.to_csv("reporte.csv", index=False)
  \end{pyblock}

  \vspace{2pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ Pipeline: \texttt{read\_csv()} $\to$ transformar
    $\to$ \texttt{to\_csv()}.
    Siempre \texttt{index=False}.}
\end{frame}
