% ─────────────────────────────────────────────────────────────────────────────
%  REPASO: REGLAS DE and / or / not
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=25]{Recordatorio: \texttt{and}, \texttt{or}, \texttt{not}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Tres operadores, tres reglas:}

  \vspace{6pt}
  {\footnotesize
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}l l l@{}}
    \toprule
    {\bfseries\color{arcaDark} Operador} &
    {\bfseries\color{arcaDark} Resultado} &
    {\bfseries\color{arcaDark} En una frase} \\
    \midrule
    \texttt{A and B} & \texttt{True} solo si \textbf{ambos} son \texttt{True}
      & ``Se necesitan las dos cosas'' \\
    \texttt{A or B}  & \texttt{True} si \textbf{al menos uno} es \texttt{True}
      & ``Basta con una de las dos'' \\
    \texttt{not A}   & Invierte el valor
      & ``Lo contrario'' \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{pyblock}
hay_stock = True
tiene_credito = False
zona_bloqueada = False

print(hay_stock and tiene_credito)
print(hay_stock or tiene_credito)
print(not zona_bloqueada)
      \end{pyblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      \vspace{4pt}
      \pause
      \begin{outputblock}
False
True
True
      \end{outputblock}
      \vspace{4pt}
      {\footnotesize\color{textMuted}
        El stock existe pero el crédito no:
        \texttt{and} exige ambos,
        \texttt{or} acepta cualquiera.
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  PRÁCTICA: CONSTRUIR CONDICIONES

\begin{frame}[fragile,shrink=30]{Práctica: Traducir Reglas a Código \hfill {\normalsize\color{arcaRed}\faClock\hspace{4pt}5 min}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Traduzcan cada regla de negocio a una expresión booleana en Python.}

  \vspace{4pt}
  \begin{pyblock}
temperatura = 38
humedad = 72
en_mantenimiento = False
stock = 45
minimo = 50
es_preferente = True
  \end{pyblock}

  \vspace{4pt}
  {\footnotesize\color{textMuted}
  \begin{enumerate}\setlength{\itemsep}{4pt}
    \item[{\color{arcaRed}\bfseries 1.}]
      ``Hay riesgo ambiental si la temperatura supera 35°C
      \textbf{o} la humedad supera 80\%.''
    \item[{\color{arcaRed}\bfseries 2.}]
      ``La alarma suena si hay riesgo
      \textbf{y} el sistema no está en mantenimiento.''
    \item[{\color{arcaRed}\bfseries 3.}]
      ``Se despacha si el stock alcanza
      \textbf{o} el cliente es preferente.''
    \item[{\color{arcaRed}\bfseries 4.}]
      ``Se bloquea si el stock no alcanza
      \textbf{y} no es preferente.''
  \end{enumerate}
  }

  \vspace{3pt}
  {\scriptsize\color{arcaDark}
    Las reglas 3 y 4 son complementarias.
    Verifiquen que una dé \texttt{True} y la otra \texttt{False}.}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  SOLUCIÓN
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{Solución --- Traducir Reglas a Código}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
temperatura = 38
humedad = 72
en_mantenimiento = False
stock = 45
minimo = 50
es_preferente = True

riesgo = temperatura > 35 or humedad > 80
alarma = riesgo and not en_mantenimiento
despacha = stock >= minimo or es_preferente
bloquea = stock < minimo and not es_preferente

print("Riesgo:", riesgo)
print("Alarma:", alarma)
print("Despacha:", despacha)
print("Bloquea:", bloquea)
      \end{pyblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{4pt}
      \begin{outputblock}
Riesgo: True
Alarma: True
Despacha: True
Bloquea: False
      \end{outputblock}
      \vspace{4pt}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{or} para riesgo: basta
          que un sensor detecte peligro.
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{and not} para la alarma:
          el mantenimiento la silencia.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Cambien \texttt{es\_preferente = False}
          y verifiquen que ahora sí se bloquea.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Listas --- Cómo Funcionan los Índices}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Una lista guarda varios valores bajo un solo nombre.
    Cada valor tiene una \textbf{posición} (índice) que empieza en \texttt{0}.}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    cell/.style={draw=arcaRed, fill=arcaGray, minimum width=2.2cm,
                 minimum height=0.6cm, font=\footnotesize\ttfamily},
    idx/.style={font=\scriptsize\color{arcaRed}\bfseries}
  ]
    \node[cell] (c0) at (0,0) {"GYE Norte"};
    \node[cell] (c1) at (2.4,0) {"GYE Sur"};
    \node[cell] (c2) at (4.8,0) {"Manta"};
    \node[cell] (c3) at (7.2,0) {"Cuenca"};
    \node[idx, above=1pt of c0] {0};
    \node[idx, above=1pt of c1] {1};
    \node[idx, above=1pt of c2] {2};
    \node[idx, above=1pt of c3] {3};
    \node[font=\scriptsize\color{textMuted}, below=1pt of c0] {-4};
    \node[font=\scriptsize\color{textMuted}, below=1pt of c1] {-3};
    \node[font=\scriptsize\color{textMuted}, below=1pt of c2] {-2};
    \node[font=\scriptsize\color{textMuted}, below=1pt of c3] {-1};
  \end{tikzpicture}
  \end{center}

  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{pyblock}
rutas = ["GYE Norte","GYE Sur","Manta","Cuenca"]
print(rutas[0])
print(rutas[2])
print(rutas[-1])
print(len(rutas))
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
GYE Norte
Manta
Cuenca
4
      \end{outputblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{[0]} es el \textbf{primer} elemento.
          No el 1.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{[-1]} es el \textbf{último},
          \texttt{[-2]} el penúltimo.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{len()} devuelve la cantidad
          de elementos (4), pero el último
          índice válido es \texttt{3}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{rutas[4]} daría error:
          \texttt{IndexError}.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  LISTAS: SLICING
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=25]{Listas --- Slicing (Rebanadas)}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Con \texttt{lista[inicio:fin]} extraemos un subconjunto.
    El índice \texttt{fin} \textbf{no se incluye}.}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    cell/.style={draw=arcaRed, fill=arcaGray, minimum width=1.6cm,
                 minimum height=0.55cm, font=\scriptsize\ttfamily},
    idx/.style={font=\tiny\color{arcaRed}\bfseries}
  ]
    \node[cell] (c0) at (0,0) {1.50};
    \node[cell] (c1) at (1.8,0) {1.25};
    \node[cell] (c2) at (3.6,0) {0.90};
    \node[cell] (c3) at (5.4,0) {1.10};
    \node[cell] (c4) at (7.2,0) {0.80};
    \node[idx, above=1pt of c0] {0};
    \node[idx, above=1pt of c1] {1};
    \node[idx, above=1pt of c2] {2};
    \node[idx, above=1pt of c3] {3};
    \node[idx, above=1pt of c4] {4};
    \draw[arcaRed, thick, decorate, decoration={brace, amplitude=4pt, mirror}]
      ([yshift=-3pt]c1.south west) -- ([yshift=-3pt]c2.south east)
      node[midway, below=4pt, font=\tiny\color{arcaRed}] {\texttt{[1:3]}};
  \end{tikzpicture}
  \end{center}

  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{pyblock}
precios = [1.50, 1.25, 0.90, 1.10, 0.80]
print("precios[1:3] =", precios[1:3])
print("precios[:2]  =", precios[:2])
print("precios[3:]  =", precios[3:])
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
precios[1:3] = [1.25, 0.9]
precios[:2]  = [1.5, 1.25]
precios[3:]  = [1.1, 0.8]
      \end{outputblock}
    \end{column}

    \begin{column}{0.46\textwidth}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{[1:3]} $\rightarrow$ índices 1 y 2.
          El 3 \textbf{queda fuera}.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{[:2]} = desde el inicio hasta
          el índice 2 (sin incluirlo).
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{[3:]} = desde el índice 3
          hasta el final.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          La lista original \textbf{no se modifica}.
          Slicing crea una copia.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=25]{Listas --- Métodos Esenciales}

\begin{frame}[fragile,shrink=30]{Ya Sabemos Acceder --- Pero Uno por Uno}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Acabamos de practicar \texttt{lista[i]} para leer elementos.
    ¿Qué pasa cuando la lista tiene muchos valores?}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      {\scriptsize\color{arcaRed}\bfseries\faTimesCircle\hspace{4pt}
        Acceso manual --- no escala}
      \begin{pyblock}
precios = [1.50, 1.25, 0.90, 1.10, 0.80]
print("Producto 1:", precios[0])
print("Producto 2:", precios[1])
print("Producto 3:", precios[2])
print("Producto 4:", precios[3])
print("Producto 5:", precios[4])
# Y si fueran 500 productos...?
      \end{pyblock}
    \end{column}

    \begin{column}{0.50\textwidth}
      {\scriptsize\color{codeGreen}\bfseries\faCheckCircle\hspace{4pt}
        Con ciclo \texttt{for} --- funciona para 5 o 5\,000}
      \begin{pyblock}
precios = [1.50, 1.25, 0.90, 1.10, 0.80]
for precio in precios:
    print("Producto:", precio)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Producto: 1.5
Producto: 1.25
Producto: 0.9
Producto: 1.1
Producto: 0.8
      \end{outputblock}
    \end{column}
  \end{columns}

  \vspace{4pt}
  {\footnotesize\color{arcaDark}
    \textbf{El ciclo \texttt{for}} toma cada elemento de la lista,
    lo asigna a una variable temporal (\texttt{precio}),
    y ejecuta el bloque indentado con ese valor.
    La misma lógica para 5 elementos o 5\,000.}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  ANATOMÍA DEL for — DIAGRAMA DE FLUJO
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=25]{Anatomía del Ciclo \texttt{for}}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.38\textwidth}
      \vspace{-2pt}
      \begin{center}
      \begin{tikzpicture}[
        node distance=0.55cm,
        startstop/.style={rounded rectangle, draw=arcaDark, fill=arcaDark!10,
                  font=\tiny\bfseries, minimum width=2.4cm,
                  minimum height=0.4cm, align=center},
        process/.style={rectangle, draw=arcaRed, fill=arcaRed!8,
                  font=\tiny\ttfamily, minimum width=2.4cm,
                  minimum height=0.4cm, align=center},
        decision/.style={diamond, draw=codeOrange, fill=codeOrange!10,
                  font=\tiny\bfseries, aspect=2.2, inner sep=1pt,
                  align=center},
        arr/.style={-{Stealth[length=2.5pt]}, line width=0.5pt}
      ]
        \node[startstop] (start) {Inicio del \texttt{for}};
        \node[decision, below=of start] (check) {¿Quedan\\elementos?};
        \node[process, below=of check] (assign) {producto = siguiente};
        \node[process, below=of assign] (body) {ejecutar bloque};
        \node[startstop, right=1.2cm of check] (end) {Fin del\\ciclo};

      \vspace{4pt}
      {\tiny\color{textMuted}
        El ciclo repite el bloque una vez por cada
        elemento. Cuando no quedan más, continúa
        con el código que sigue después.}
    \end{column}

    \begin{column}{0.58\textwidth}
      {\footnotesize\color{arcaDark}\bfseries La sintaxis:}
      \vspace{2pt}
      \begin{center}

                    font=\scriptsize\ttfamily\bfseries, inner xsep=4pt,
                    minimum height=0.4cm},
      ]
        \node[font=\scriptsize\ttfamily\color{arcaRed}\bfseries] (for) {for};
        \node[box, right=2pt of for] (var) {\color{arcaDark}producto};

        \node[box, right=2pt of in, draw=arcaDark, fill=arcaDark!8]
          (lst) {\color{arcaDark}productos};
        \node[font=\scriptsize\ttfamily, right=1pt of lst] (col) {:};
        \node[lbl, below=2pt of var] {variable\\temporal};
        \node[lbl, below=2pt of lst] {secuencia\\a recorrer};
      \end{tikzpicture}
      \end{center}

      \vspace{4pt}
      \begin{pyblock}
productos = ["Coca-Cola", "Sprite", "Dasani"]
for producto in productos:

      \end{outputblock}

      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{producto} toma un valor distinto en cada vuelta.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Solo el código \textbf{indentado} se repite.
          \texttt{print("Listo")} se ejecuta una sola vez al final.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          El nombre de la variable es libre.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Entendiendo \texttt{range(inicio, fin, paso)}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries

  \vspace{4pt}

  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}l l l@{}}

          (\texttt{total += venta}).

          \textbf{Contador:} suma 1
          (\texttt{cont += 1}).
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Se pueden usar \textbf{varios}
          contadores en un mismo ciclo.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]

  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    4 productos despachados hoy. Calculen ingreso por producto
    y acumulen el total.}

  \vspace{4pt}
  \begin{pyblock}
productos = ["Coca-Cola 600ml","Sprite 500ml","Dasani 1L","Fanta 400ml"]
unidades  = [1200, 800, 2000, 500]
precios   = [1.50, 1.25, 0.90, 1.10]
  \end{pyblock}

  \vspace{3pt}
  {\footnotesize\color{textMuted}
    \textbf{Tarea:}
    \begin{enumerate}\setlength{\itemsep}{2pt}
      \item Calculen el ingreso de cada producto:
        \texttt{ingreso = unidades[i] * precios[i]}
      \item Acumulen el ingreso total del día.
      \item Marquen con \texttt{``SOBRE 1000''} los que superen \$\,1\,000.
    \end{enumerate}
  }
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  SOLUCIÓN REPORTE
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{Solución --- Reporte de Ingresos}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{pyblock}
productos = ["Coca-Cola 600ml","Sprite 500ml",
             "Dasani 1L","Fanta 400ml"]
unidades  = [1200, 800, 2000, 500]
precios   = [1.50, 1.25, 0.90, 1.10]

total = 0
for i in range(len(productos)):
    ingreso = unidades[i] * precios[i]
    total += ingreso
    etiqueta = ""
    if ingreso >= 1000:
        etiqueta = "SOBRE 1000"
    print(productos[i], "->", ingreso, etiqueta)

print("Total del dia:", total)
      \end{pyblock}
    \end{column}

    \begin{column}{0.42\textwidth}
      \vspace{4pt}
      \begin{outputblock}
Coca-Cola 600ml -> 1800.0 SOBRE 1000
Sprite 500ml -> 1000.0 SOBRE 1000
Dasani 1L -> 1800.0 SOBRE 1000
Fanta 400ml -> 550.0
Total del dia: 5150.0
      \end{outputblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=35]{El Operador \texttt{in} --- Buscar dentro de una lista}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    \texttt{in} responde ``¿este valor está en la lista?''
    con \texttt{True} o \texttt{False}.}

  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{pyblock}
frutas = ["manzana", "pera", "uva"]
print("pera" in frutas)
print("kiwi" in frutas)
print("kiwi" not in frutas)
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
True
False
True
      \end{outputblock}

      \vspace{3pt}
      {\footnotesize\color{arcaDark}\bfseries Uso práctico con \texttt{if}:}
      \vspace{2pt}
      \begin{pyblock}
zonas = ["Norte", "Sur", "Centro"]
pedido = "Oriente"
if pedido in zonas:
    print("Zona valida")
else:
    print("Zona NO habilitada")
      \end{pyblock}
      \vspace{-2pt}
      \begin{outputblock}
Zona NO habilitada
      \end{outputblock}
    \end{column}

    \begin{column}{0.44\textwidth}
      \vspace{2pt}
      {\footnotesize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{x in lista} $\rightarrow$
          \texttt{True} si \texttt{x} aparece.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          \texttt{x not in lista} $\rightarrow$
          \texttt{True} si \textbf{no} está.
        \item[{\color{arcaRed}\scriptsize\faArrowRight}]
          Ideal dentro de un \texttt{if}
          para validar pertenencia a
          un grupo.
        \item[{\color{arcaRed}\scriptsize\faLightbulb}]
          Funciona con strings:
          \texttt{"py" in "python"} es
          \texttt{True}.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=30]{Práctica: Validador de Pedidos \hfill {\normalsize\color{arcaRed}\faClock\hspace{4pt}5 min}}
  \vspace{-6pt}
  {\small\color{arcaDark}\bfseries
    Un restaurante solo prepara ciertos platos.
    Verifiquen cuáles pedidos se pueden atender.}

  \vspace{4pt}
  \begin{pyblock}
menu = ["hamburguesa", "pizza", "ensalada", "tacos"]
pedidos = ["pizza", "sushi", "tacos", "pasta", "ensalada"]
  \end{pyblock}

  \vspace{3pt}
  {\footnotesize\color{textMuted}
    \textbf{Tarea:}
    \begin{enumerate}\setlength{\itemsep}{2pt}
      \item Recorrer \texttt{pedidos} con un \texttt{for}.
      \item Usar \texttt{in} para verificar si el pedido está en el \texttt{menu}.
      \item Si está: \texttt{"Preparando: ..."}. Si no: \texttt{"No disponible: ..."}.
      \item Contar cuántos se rechazaron.
    \end{enumerate}
  }

  \vspace{3pt}
  {\footnotesize\color{arcaDark}\bfseries Salida esperada:}
  \begin{outputblock}
Preparando: pizza
No disponible: sushi
Preparando: tacos
No disponible: pasta
Preparando: ensalada
Rechazados: 2
  \end{outputblock}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
%  SOLUCIÓN in
% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}[fragile,shrink=30]{Solución --- Validador de Pedidos}
  \vspace{-6pt}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{pyblock}
menu = ["hamburguesa", "pizza",
        "ensalada", "tacos"]
pedidos = ["pizza", "sushi", "tacos",
           "pasta", "ensalada"]
rechazados = 0

for plato in pedidos:
    if plato in menu:
        print("Preparando:", plato)
    else:
        print("No disponible:", plato)
        rechazados += 1

print("Rechazados:", rechazados)
      \end{pyblock}
    \end{column}

    \begin{column}{0.42\textwidth}
      \vspace{4pt}
      \begin{outputblock}
Preparando: pizza
No disponible: sushi
Preparando: tacos
No disponible: pasta
Preparando: ensalada
Rechazados: 2
      \end{outputblock}
      \vspace{4pt}
      {\scriptsize\color{textMuted}
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{for plato in pedidos} recorre
          directamente los valores.
        \item[{\color{codeGreen}\scriptsize\faCheck}]
          \texttt{plato in menu} verifica
          pertenencia sin índices.
      \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}
