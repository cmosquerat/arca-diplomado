\begin{frame}[shrink=5]{Métodos de Detección: IQR vs Z-Score}
  \vspace{-4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codeBlue} Método IQR}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item Límite inferior: $Q1 - 1.5 \times IQR$
        \item Límite superior: $Q3 + 1.5 \times IQR$
        \item \textbf{Ventaja:} Robusto --- no le afectan los propios outliers.
        \item \textbf{Cuándo:} Datos sesgados, distribuciones no normales.
        \item \textbf{En Wine Quality:} \texttt{residual\_sugar}, \texttt{chlorides}.
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codePurple} Método Z-Score}
      \vspace{4pt}

      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item Outlier si $|z| > 3$ (o 2.5, depende del criterio).
        \item $z = \frac{x - \mu}{\sigma}$
        \item \textbf{Ventaja:} Intuitivo --- ``cuántas desviaciones del centro''.
        \item \textbf{Cuándo:} Datos aproximadamente normales.
        \item \textbf{En Wine Quality:} \texttt{pH}, \texttt{density}.
      \end{itemize}
      }
    \end{column}
  \end{columns}

  \vspace{8pt}
  \begin{center}
  \begin{tikzpicture}
    \node[rectangle, rounded corners=4pt, draw=codeOrange, fill=codeOrange!8,
          minimum width=12cm, minimum height=0.8cm,
          font=\scriptsize\bfseries\color{codeOrange}, align=center] {
      \faExclamationTriangle\ Ambos métodos son heurísticas.
      No existe un método ``perfecto'' --- siempre requieren criterio humano.
    };
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}[fragile,shrink=28]{Detectar Outliers con IQR en Wine Quality}
  \vspace{-6pt}
  \begin{pyblock}
col = "residual sugar"
Q1 = df[col].quantile(0.25)
Q3 = df[col].quantile(0.75)
IQR = Q3 - Q1

lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR

outliers_iqr = df[(df[col] < lower) | (df[col] > upper)]
print(f"Limites: [{lower:.2f}, {upper:.2f}]")
print(f"Outliers encontrados: {len(outliers_iqr)} "
      f"de {len(df)} ({len(outliers_iqr)/len(df)*100:.1f}%)")
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ El azúcar residual tiene una distribución muy sesgada a la derecha.
    La mayoría de los vinos tienen poca azúcar ($\sim$2 g/L),
    pero algunos llegan a 15.5 g/L.}
\end{frame}

\begin{frame}[shrink=10]{Z-Score: Desglosando la Fórmula}
  \vspace{-4pt}
  {\footnotesize
    El Z-score responde una pregunta simple:
    \textbf{``¿A cuántas desviaciones estándar está este dato del centro?''}}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}
    \node[rectangle, rounded corners=4pt, draw=arcaRed, fill=arcaRed!6,
          minimum width=5cm, minimum height=1.5cm,
          font=\Large\color{arcaDark}, align=center] (formula) {
      $z = \dfrac{x - \mu}{\sigma}$
    };
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  {\small
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{@{}clp{7.5cm}@{}}
    \toprule
    \textbf{Símbolo} & \textbf{Nombre} & \textbf{Significado} \\
    \midrule
    $x$ & El dato & El valor individual que estamos evaluando. \\
    $\mu$ & Media ($\bar{x}$) & El ``centro'' de todos los datos. \\
    $\sigma$ & Desviación estándar & La ``unidad de dispersión'' --- qué tan lejos se esparcen los datos
      típicamente alrededor de $\mu$. \\
    $x - \mu$ & Desviación & Qué tan lejos está $x$ del centro (en unidades originales). \\
    $z$ & Z-score & Esa misma distancia, pero expresada en \textbf{unidades de $\sigma$}.
      Es un número \textbf{sin unidades}. \\
    \bottomrule
  \end{tabular}
  }
\end{frame}

\begin{frame}[shrink=10]{Z-Score: Ejemplo Numérico Paso a Paso}
  \vspace{-4pt}
  {\footnotesize
    Supongamos que el \textbf{pH} del vino tiene $\mu = 3.31$ y $\sigma = 0.15$.
    Un vino tiene pH $= 3.90$. ¿Es un outlier?}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    step/.style={rectangle, rounded corners=3pt, draw=#1, fill=#1!8,
                 minimum width=12cm, minimum height=0.9cm,
                 font=\small\color{textDark}, align=left, text width=11.5cm,
                 inner sep=6pt},
  ]
    \node[step=codeBlue] (s1) at (0, 0)
      {\textbf{Paso 1:} Calcular la desviación: $x - \mu = 3.90 - 3.31 = 0.59$};
    \node[step=codePurple, below=0.3cm of s1] (s2)
      {\textbf{Paso 2:} Dividir por $\sigma$: $z = \frac{0.59}{0.15} = 3.93$};
    \node[step=arcaRed, below=0.3cm of s2] (s3)
      {\textbf{Paso 3:} Interpretar: $|z| = 3.93 > 3$ $\to$ \textbf{es outlier}};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{4pt}
    \item[{\color{codeBlue}\scriptsize\faArrowRight}]
      $z = 1$: el dato está a 1 desviación del centro. \textbf{Normal} (68\% de datos están aquí).
    \item[{\color{codeOrange}\scriptsize\faArrowRight}]
      $z = 2$: está a 2 desviaciones. \textbf{Poco común} (solo 5\% están más lejos).
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      $z = 3$: está a 3 desviaciones. \textbf{Muy raro} (solo 0.3\% están más lejos).
    \item[{\color{arcaRed}\scriptsize\faExclamationTriangle}]
      $z = 3.93$: está a casi 4 desviaciones. Este pH \textbf{no es típico} --- hay que investigar.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=5]{¿Qué Es una Heurística?}
  \vspace{-4pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=12cm, minimum height=1.3cm,
                font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                inner sep=6pt},
  ]
    \node[box=codeBlue] (def) at (0, 0)
      {\textbf{\color{codeBlue}\normalsize Definición}\\[2pt]
       Una \textbf{heurística} es una regla práctica que funciona ``suficientemente bien''
       en la mayoría de los casos, pero \textbf{no garantiza} el resultado correcto siempre.
       Es un atajo útil, no una verdad matemática.};
    \node[box=codeGreen, below=0.4cm of def] (ej)
      {\textbf{\color{codeGreen}\normalsize Ejemplos de heurísticas}\\[2pt]
       ``Si $|z| > 3$, es outlier'' $\to$ funciona bien con datos normales, pero no con datos sesgados.\\
       ``Si $Q < Q1 - 1.5 \times IQR$, es outlier'' $\to$ el 1.5 es una convención, no una ley.\\
       ``Si $|skew| > 1$, usar mediana'' $\to$ buen criterio general, pero hay excepciones.};
    \node[box=arcaRed, below=0.4cm of ej] (imp)
      {\textbf{\color{arcaRed}\normalsize ¿Por qué importa saber esto?}\\[2pt]
       Para no aplicar reglas ciegamente.
       La heurística te da un \textbf{punto de partida}; el \textbf{criterio de negocio} te da la decisión final.
       ``El algoritmo me dijo que es outlier'' no es justificación suficiente para eliminar un dato.};
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}[fragile,shrink=28]{Detectar Outliers con Z-Score en Código}

# Calcular Z-score de cada dato

# Ver algunos ejemplos
print("Primeros 5 z-scores:")
for i in range(5):
    print(f"  pH={df[col].iloc[i]:.2f}  ->  "
          f"z={z_scores[i]:.2f}")

# Detectar outliers
threshold = 3
outliers_z = df[np.abs(z_scores) > threshold]
print(f"\nOutliers con |z| > {threshold}: {len(outliers_z)}")
print(f"Valores de pH extremos: {outliers_z[col].values}")
  \end{pyblock}

    \faLightbulb\ El pH tiene distribución casi normal,
    así que el Z-score funciona bien aquí.
    Para \texttt{residual sugar} (sesgada), prefiere IQR.}

\begin{frame}[shrink=10]{Técnicas de Tratamiento de Outliers}
  \vspace{-4pt}
  {\small
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}p{2.8cm}p{4.5cm}p{4.5cm}@{}}
    \toprule
    \textbf{Técnica} & \textbf{Qué hace} & \textbf{Cuándo usarla} \\
    \midrule
    \textbf{Eliminar} &
      Quita las filas con outliers &
      Error confirmado. Pocas filas. \\
    \textbf{Winsorizar} &
      Recorta los extremos al percentil 5/95 &
      Quieres mantener la fila pero limitar el impacto. \\
    \textbf{Capping (IQR)} &
      Reemplaza por el límite IQR &
      Similar a winsorizar, basado en cuartiles. \\
    \textbf{Transformar (log)} &
      Aplica $\log(x)$ para comprimir la escala &
      Distribución muy sesgada a la derecha. \\
    \textbf{Imputar} &
      Reemplaza por mediana o media &
      Si tratas el outlier como ``dato faltante''. \\
    \textbf{Mantener} &
      No hacer nada &
      El outlier es información real y valiosa. \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ No hay una técnica ``correcta''. Depende del problema, los datos y el modelo.}
\end{frame}

\sectionframe{La Distribución Normal}{¿Por qué importa tanto y cómo verificarla?}

\begin{frame}[shrink=5]{¿Qué Es la Distribución Normal?}
  \vspace{-4pt}
  {\footnotesize
    La distribución Normal (o \textbf{Gaussiana}) es la distribución más importante de la estadística.
    Es la famosa ``campana'' --- simétrica, con la mayoría de datos cerca del centro y pocos en los extremos.}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}
    \node[rectangle, rounded corners=4pt, draw=arcaRed, fill=arcaRed!6,
          minimum width=6cm, minimum height=1.5cm,
          font=\large\color{arcaDark}, align=center] (formula) {
      $X \sim N(\mu, \sigma^2)$
    };
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  {\footnotesize Se lee: ``$X$ sigue una distribución Normal con media $\mu$ y varianza $\sigma^2$''.}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{4pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \textbf{¿Por qué ``Normal''?} Porque aparece en una cantidad enorme de fenómenos naturales:
      estaturas, errores de medición, presión arterial, temperaturas, pesos de productos.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \textbf{¿Por qué es tan común?} Por el \textbf{Teorema del Límite Central}: cuando muchos
      factores pequeños e independientes se suman, el resultado tiende a ser Normal.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      \textbf{Ejemplo:} La estatura depende de genética, nutrición, salud, etc.
      Ningún factor domina $\to$ la suma tiende a Normal.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=10]{Los Dos Parámetros: $\mu$ y $\sigma$}
  \vspace{-4pt}
  {\footnotesize
    La distribución Normal se define \textbf{completamente} con solo dos números:}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{center}
      \begin{tikzpicture}[
        box/.style={rectangle, rounded corners=4pt, draw=codeBlue, fill=codeBlue!8,
                    minimum width=5.5cm, minimum height=2.8cm,
                    font=\scriptsize\color{textDark}, align=left, text width=5cm,
                    inner sep=6pt},
      ]
        \node[box] {
          {\normalsize\bfseries\color{codeBlue} $\mu$ (mu) --- La Media}\\[4pt]
          Es el \textbf{centro} de la campana.\\[2pt]
          Determina \textbf{dónde} se ubica la distribución en el eje X.\\[2pt]
          Si $\mu$ cambia, la campana se \textbf{desplaza} a la izquierda o derecha,
          pero no cambia su forma.\\[2pt]
          {\color{textMuted}Ej: pH con $\mu = 3.31$ está centrado en 3.31.}
        };
      \end{tikzpicture}
      \end{center}
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{center}
      \begin{tikzpicture}[
        box/.style={rectangle, rounded corners=4pt, draw=codePurple, fill=codePurple!8,
                    minimum width=5.5cm, minimum height=2.8cm,
                    font=\scriptsize\color{textDark}, align=left, text width=5cm,
                    inner sep=6pt},
      ]
        \node[box] {
          {\normalsize\bfseries\color{codePurple} $\sigma$ (sigma) --- La Desv. Estándar}\\[4pt]
          Es el \textbf{ancho} de la campana.\\[2pt]
          Determina qué tan \textbf{dispersos} están los datos alrededor de $\mu$.\\[2pt]
          $\sigma$ pequeño $\to$ campana alta y estrecha (datos muy concentrados).\\
          $\sigma$ grande $\to$ campana baja y ancha (datos dispersos).\\[2pt]
          {\color{textMuted}Ej: pH con $\sigma = 0.15$ es bastante concentrado.}
        };
      \end{tikzpicture}
      \end{center}
    \end{column}
  \end{columns}

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ Con solo $\mu$ y $\sigma$, puedes saber \textbf{todo} sobre una distribución Normal:
    dónde está el centro, qué tan dispersa es, y qué porcentaje de datos cae en cualquier rango.}
\end{frame}

\begin{frame}[shrink=5]{Propiedades Clave de la Normal}
  \vspace{-4pt}
  \begin{center}
  \begin{tikzpicture}[
    prop/.style={rectangle, rounded corners=4pt, draw=arcaRed, fill=arcaRed!6,
                 minimum width=5.5cm, minimum height=1.3cm,
                 font=\scriptsize\color{textDark}, align=center, text width=5cm},
  ]
    \node[prop] (p1) at (0, 0)
      {\textbf{\color{arcaRed}Perfectamente simétrica}\\[2pt]
       $\text{media} = \text{mediana} = \text{moda}$\\
       Skewness $= 0$, Kurtosis $= 3$};
    \node[prop] (p2) at (6.5, 0)
      {\textbf{\color{arcaRed}Regla 68--95--99.7}\\[2pt]
       68\% dentro de $\mu \pm 1\sigma$\\
       95\% dentro de $\mu \pm 2\sigma$\\
       99.7\% dentro de $\mu \pm 3\sigma$};
    \node[prop] (p3) at (0, -2.2)
      {\textbf{\color{arcaRed}Colas infinitas}\\[2pt]
       Teóricamente va de $-\infty$ a $+\infty$\\
       pero en la práctica casi todo\\
       está dentro de $\pm 3\sigma$};
    \node[prop] (p4) at (6.5, -2.2)
      {\textbf{\color{arcaRed}La estandarización}\\[2pt]
       Cualquier Normal se convierte\\
       en $N(0, 1)$ con el Z-score:\\
       $z = (x - \mu) / \sigma$};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize\color{textMuted}
    \faArrowRight\ La \textbf{Normal estándar} $N(0, 1)$ tiene $\mu = 0$ y $\sigma = 1$.
    Es la ``versión universal'' --- permite comparar datos de cualquier escala.}
\end{frame}

\begin{frame}[shrink=10]{¿Por Qué Importa Saber si los Datos Son Normales?}
  \vspace{-4pt}
  {\footnotesize
    Varias herramientas que ya conocemos \textbf{asumen} que los datos son normales.
    Si no lo verificas, tus conclusiones pueden ser \textbf{incorrectas}:}

  \vspace{6pt}
  {\small
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}p{4.2cm}p{3.5cm}p{4.2cm}@{}}
    \toprule
    \textbf{Herramienta} & \textbf{Asume normalidad} & \textbf{Si no es Normal\ldots} \\
    \midrule
    Z-score para outliers &
      Sí &
      Detecta de más o de menos \\
    Regla 68--95--99.7 &
      Sí &
      Los porcentajes no aplican \\
    $\mu \pm 2\sigma$ como rango &
      Sí &
      Rango incorrecto \\
    Correlación de Pearson &
      Funciona mejor &
      Puede subestimar la relación \\
    Media como resumen &
      Más útil si Normal &
      Mediana es mejor opción \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faExclamationTriangle\ No es ``bonito'' que los datos sean normales.
    Es un \textbf{requisito} para que ciertas herramientas funcionen correctamente.}
\end{frame}

\begin{frame}[shrink=5]{Tres Herramientas para Verificar Normalidad}

\begin{frame}[shrink=10]{Q-Q Plot: ¿Qué Es y Cómo Leerlo?}
  \vspace{-4pt}
  {\footnotesize
    Q-Q = \textbf{Quantile-Quantile}. Compara los cuantiles de \textbf{tus datos reales}
    con los cuantiles que \textbf{tendrían si fueran perfectamente normales}.}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    step/.style={rectangle, rounded corners=3pt, draw=#1, fill=#1!8,
                 minimum width=12cm, minimum height=0.85cm,
                 font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                 inner sep=6pt},
  ]
    \node[step=codeBlue] (s1) at (0, 0)
      {\textbf{Paso 1:} Ordena tus $n$ datos de menor a mayor.
       El más pequeño es el cuantil 1/$n$, el siguiente 2/$n$, etc.};
    \node[step=codePurple, below=0.25cm of s1] (s2)
      {\textbf{Paso 2:} Calcula los cuantiles teóricos: ``si estos datos fueran
       perfectamente normales, ¿qué valor tendría el cuantil 1/$n$, 2/$n$, etc.?''};
    \node[step=codeGreen, below=0.25cm of s2] (s3)
      {\textbf{Paso 3:} Dibuja un punto por cada dato: eje X $=$ cuantil teórico,
       eje Y $=$ dato real. Si los datos son normales, los puntos caen sobre la diagonal.};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{4pt}
    \item[{\color{codeGreen}\scriptsize\faCheckCircle}]
      \textbf{Puntos sobre la línea:} los datos son normales --- cada cuantil real
      coincide con el teórico.
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      \textbf{Puntos se curvan arriba en los extremos:} cola derecha más pesada
      (más valores extremos altos de los esperados).
    \item[{\color{arcaRed}\scriptsize\faTimesCircle}]
      \textbf{Forma de S:} distribución sesgada --- los datos \textbf{no son normales}.
  \end{itemize}
  }
\end{frame}

% ── Shapiro-Wilk en detalle ──
\begin{frame}[shrink=10]{Shapiro-Wilk: El ``Detector de Normalidad''}
  \vspace{-4pt}
  {\footnotesize
    El test de Shapiro-Wilk responde una pregunta directa:
    \textbf{``¿Estos datos podrían venir de una distribución Normal?''}}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=12cm, minimum height=1.3cm,
                font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                inner sep=6pt},
  ]
    \node[box=codeBlue] (idea) at (0, 0)
      {\textbf{\color{codeBlue}\faLightbulb\ La idea intuitiva:}\\
       Shapiro-Wilk ordena los datos y compara las distancias entre ellos con las
       distancias que \textbf{esperaríamos} si fueran normales. Si coinciden bien,
       el resultado (llamado \textbf{estadístico $W$}) es cercano a \textbf{1}.};
    \node[box=codeGreen, below=0.3cm of idea] (si)
      {\textbf{\color{codeGreen}\faCheckCircle\ $W$ cercano a 1} (ej: $W = 0.997$)\\
       Los datos se comportan como una Normal. Las distancias entre valores
       consecutivos coinciden con lo esperado. \textbf{Conclusión: parece Normal.}};
    \node[box=arcaRed, below=0.3cm of si] (no)
      {\textbf{\color{arcaRed}\faTimesCircle\ $W$ lejos de 1} (ej: $W = 0.85$)\\
       Las distancias entre datos no coinciden con una Normal. Hay
       asimetría, colas pesadas u otra forma distinta. \textbf{Conclusión: NO es Normal.}};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{codePurple}\scriptsize\faArrowRight}]
      En Python: \texttt{stat, p = stats.shapiro(datos)}.
      El valor \texttt{stat} es $W$.
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      \textbf{Limitación:} con datasets muy grandes ($n > 5000$), Shapiro casi siempre
      dice ``no es Normal'' porque detecta desviaciones mínimas. Combínalo con el Q-Q plot.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=5]{¿Qué Es Correlación?}
  \vspace{-4pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=3.5cm, minimum height=1.8cm,
                font=\scriptsize\color{textDark}, align=center, text width=3.2cm},
  ]
    \node[box=codeGreen] (pos) at (0,0)
      {\textbf{\color{codeGreen}Correlación positiva}\\[2pt]
       $r \approx +1$\\
       Una sube $\to$ la otra sube\\
       {\tiny Ej: alcohol $\uparrow$ quality $\uparrow$}};
    \node[box=textMuted] (zero) at (4.5,0)
      {\textbf{\color{textMuted}Sin correlación lineal}\\[2pt]
       $r \approx 0$\\
       No hay relación \textbf{lineal}\\
       {\tiny ¡Puede haber otra relación!}};
    \node[box=arcaRed] (neg) at (9,0)
      {\textbf{\color{arcaRed}Correlación negativa}\\[2pt]
       $r \approx -1$\\
       Una sube $\to$ la otra baja\\
       {\tiny Ej: pH $\uparrow$ acidez $\downarrow$}};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      La correlación mide la \textbf{fuerza y dirección} de una relación \textbf{lineal} entre dos variables.
    \item[{\color{arcaRed}\scriptsize\faArrowRight}]
      Rango: de $-1$ (perfecta negativa) a $+1$ (perfecta positiva).
    \item[{\color{codeOrange}\scriptsize\faExclamationTriangle}]
      \textbf{$r \approx 0$ no significa ``no hay relación''} --- significa que no hay relación \textbf{lineal}.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=10]{Pearson ($r$): De Dónde Sale la Fórmula}
  \vspace{-4pt}
  {\footnotesize
    Queremos un número que diga: ``cuando $x$ sube, ¿$y$ también sube?''
    Construyamos la idea paso a paso:}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    step/.style={rectangle, rounded corners=3pt, draw=#1, fill=#1!8,
                 minimum width=12cm, minimum height=0.85cm,
                 font=\scriptsize\color{textDark}, align=left, text width=11.5cm,
                 inner sep=6pt},
  ]
    \node[step=codeBlue] (s1) at (0, 0)
      {\textbf{1. Desviación de cada dato:} $(x_i - \bar{x})$ mide cuánto se aleja $x_i$
       de su media. Igual para $y$: $(y_i - \bar{y})$.};
    \node[step=codePurple, below=0.2cm of s1] (s2)
      {\textbf{2. Producto de desviaciones:} $(x_i - \bar{x})(y_i - \bar{y})$.
       Si ambos están \textbf{arriba} de su media $\to$ producto \textbf{positivo}.
       Si uno arriba y otro abajo $\to$ producto \textbf{negativo}.};
    \node[step=codeGreen, below=0.2cm of s2] (s3)
      {\textbf{3. Sumar todos los productos:} $\sum(x_i - \bar{x})(y_i - \bar{y})$.
       Si la mayoría son positivos $\to$ relación positiva. Si negativos $\to$ negativa.};
    \node[step=arcaRed, below=0.2cm of s3] (s4)
      {\textbf{4. Normalizar:} Dividimos entre $(n-1)\, s_x \, s_y$ para que $r$
       quede entre $-1$ y $+1$, sin importar las unidades originales.};
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}
    \node[rectangle, rounded corners=4pt, draw=arcaRed, fill=arcaRed!6,
          minimum width=6cm, minimum height=1.2cm,
          font=\large\color{arcaDark}, align=center] {
      $r = \dfrac{\sum_{i=1}^{n}(x_i - \bar{x})(y_i - \bar{y})}{(n-1)\, s_x \, s_y}$
    };
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}[shrink=10]{Spearman ($\rho$): Pearson Pero con Rangos}
  \vspace{-4pt}
  {\footnotesize
    Spearman tiene una idea elegante: en vez de usar los \textbf{valores} originales,
    usa sus \textbf{posiciones} (rangos) cuando los ordenas de menor a mayor.}

  \vspace{4pt}
  \begin{center}
  {\small
  \renewcommand{\arraystretch}{1.2}
  \begin{tabular}{@{}cccc@{}}
    \toprule
    \textbf{Dato $x$} & \textbf{Rango de $x$} & \textbf{Dato $y$} & \textbf{Rango de $y$} \\
    \midrule
    3.2 & 1 & 5 & 1 \\
    4.1 & 2 & 6 & 2 \\
    8.7 & 3 & 7 & 3 \\
    150.0 {\tiny(outlier!)} & 4 & 8 & 4 \\
    \bottomrule
  \end{tabular}
  }
  \end{center}

  \vspace{4pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=5.5cm, minimum height=1.6cm,
                font=\scriptsize\color{textDark}, align=left, text width=5cm,
                inner sep=6pt},
  ]
    \node[box=codePurple] (how) at (0,0)
      {\textbf{\color{codePurple}¿Cómo se calcula?}\\[2pt]
       Reemplaza cada valor por su rango (posición al ordenar)
       y luego aplica la \textbf{fórmula de Pearson} sobre esos rangos.};
    \node[box=codeGreen] (why) at (6.5,0)
      {\textbf{\color{codeGreen}¿Por qué funciona?}\\[2pt]
       El outlier 150.0 solo tiene rango 4. No ``arrastra'' el resultado.
       Spearman mide si el \textbf{orden} se mantiene, no las magnitudes.};
  \end{tikzpicture}
  \end{center}

  \vspace{4pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ Tip: si Pearson y Spearman dan resultados muy diferentes,
    probablemente hay \textbf{outliers} o una relación \textbf{no lineal}.}
\end{frame}

\begin{frame}[shrink=5]{Pearson vs Spearman: Resumen}
  \vspace{-4pt}
  {\small
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}p{3cm}p{4.5cm}p{4.5cm}@{}}
    \toprule
    & \textbf{Pearson ($r$)} & \textbf{Spearman ($\rho$)} \\
    \midrule
    \textbf{Mide} &
      Relación \textbf{lineal} &
      Relación \textbf{monótona} \\
    \textbf{Usa} &
      Los \textbf{valores} originales &
      Los \textbf{rangos} (posiciones) \\
    \textbf{Outliers} &
      Sensible (un extremo lo distorsiona) &
      Robusto (el rango no cambia mucho) \\
    \textbf{Normalidad} &
      Funciona mejor con datos normales &
      No la requiere \\
    \textbf{Cuándo usar} &
      Datos normales, sin outliers, relación recta &
      Datos sesgados, outliers, relación curva \\
    \bottomrule
  \end{tabular}
  }

  \vspace{6pt}
  {\scriptsize\color{textMuted}
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{codeBlue}\scriptsize\faArrowRight}]
      En Python: \texttt{df["x"].corr(df["y"])} usa Pearson por defecto.
    \item[{\color{codePurple}\scriptsize\faArrowRight}]
      Para Spearman: \texttt{df["x"].corr(df["y"], method="spearman")}.
  \end{itemize}
  }
\end{frame}

\sectionframe{No Todo Es una Recta}{Relaciones no lineales entre variables}

\begin{frame}[shrink=8]{¿Por Qué Pearson Puede Fallar?}
  \vspace{-4pt}
  {\footnotesize
    Pearson mide relación \textbf{lineal}. Si la relación es curva,
    Pearson dirá ``no hay relación'' cuando \textbf{sí la hay}:}

  \vspace{6pt}
  \begin{center}
  \begin{tikzpicture}[
    box/.style={rectangle, rounded corners=4pt, draw=#1, fill=#1!8,
                minimum width=3.8cm, minimum height=2cm,
                font=\scriptsize\color{textDark}, align=center, text width=3.5cm},
  ]
    \node[box=codeGreen] (lin) at (0,0)
      {\textbf{\color{codeGreen}Lineal}\\[2pt]
       $y = a + bx$\\
       Pearson funciona bien\\
       $r = 0.95$};
    \node[box=codeOrange] (quad) at (4.8,0)
      {\textbf{\color{codeOrange}Cuadrática}\\[2pt]
       $y = a + bx + cx^2$\\
       Pearson dice $r \approx 0$\\
       ¡Pero la relación existe!};
    \node[box=codePurple] (log) at (9.6,0)
      {\textbf{\color{codePurple}Logarítmica}\\[2pt]
       $y = a + b \ln(x)$\\
       Pearson subestima\\
       Spearman captura mejor};
  \end{tikzpicture}
  \end{center}

  \vspace{6pt}
  {\scriptsize
  \begin{itemize}\setlength{\itemsep}{3pt}
    \item[{\color{arcaRed}\scriptsize\faExclamationTriangle}]
      \textbf{Moraleja:} Siempre haz el scatter plot \textbf{antes} de confiar en el número $r$.
    \item[{\color{codeBlue}\scriptsize\faArrowRight}]
      Spearman detecta relaciones monótonas (siempre sube o siempre baja),
      aunque no sean lineales.
    \item[{\color{codePurple}\scriptsize\faArrowRight}]
      Para relaciones complejas (U, olas), se necesitan técnicas más avanzadas.
  \end{itemize}
  }
\end{frame}

\begin{frame}[shrink=10]{¿Qué Hace \texttt{regplot}? --- La Matemática Detrás}
  \vspace{-4pt}
  {\footnotesize
    \texttt{sns.regplot()} dibuja los puntos \textbf{y además ajusta una línea}.
    ¿De dónde sale esa línea? Tiene dos modos:}

  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codeBlue} Modo lineal (por defecto)}
      \vspace{4pt}

      {\scriptsize
      Busca la recta $y = a + bx$ que \textbf{minimiza} la suma de errores al cuadrado:
      }

      \vspace{4pt}
      \begin{center}
      \begin{tikzpicture}
        \node[rectangle, rounded corners=3pt, draw=codeBlue, fill=codeBlue!6,
              font=\small\color{arcaDark}, inner sep=6pt] {
          $\min \displaystyle\sum_{i=1}^{n} (y_i - a - bx_i)^2$
        };
      \end{tikzpicture}
      \end{center}

      \vspace{4pt}
      {\scriptsize
      Cada $(y_i - a - bx_i)$ es la distancia vertical del punto a la recta.
      Al elevar al cuadrado y sumar, se penalizan los puntos lejanos.
      La recta que \textbf{menos error total} tenga es la elegida.
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\bfseries\color{codePurple} Modo LOWESS (\texttt{lowess=True})}
      \vspace{4pt}

      {\scriptsize
      En vez de \textbf{una} recta para todos los datos, ajusta \textbf{muchas rectas locales}:
      }

      \vspace{4pt}
      {\scriptsize
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item Para cada punto, toma los vecinos cercanos.
        \item Ajusta una recta solo con esos vecinos.
        \item El resultado: una \textbf{curva suave} que sigue la tendencia local.
      \end{itemize}
      }

      \vspace{4pt}
      {\scriptsize\color{textMuted}
      Útil cuando la relación \textbf{no es una recta}.
      No asume ninguna forma específica.
      }
    \end{column}
  \end{columns}

  \vspace{6pt}
  {\scriptsize\color{arcaDark}\bfseries
    \faArrowRight\ El modo lineal es el \textbf{método de mínimos cuadrados},
    una de las técnicas más fundamentales de la estadística.}
\end{frame}

\begin{frame}[fragile,shrink=28]{Visualizar Relaciones con regplot en Código}
  \vspace{-6pt}
  \begin{pyblock}
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

# Lineal: alcohol vs quality
sns.regplot(x="alcohol", y="quality", data=df,
            scatter_kws={"alpha": 0.2}, ax=axes[0],
            color="#16A34A", line_kws={"color": "#C82B40"})
axes[0].set_title("alcohol vs quality (lineal)")

# No lineal: citric acid vs volatile acidity
sns.regplot(x="citric acid", y="volatile acidity",
            data=df, scatter_kws={"alpha": 0.2},
            ax=axes[1], color="#2563EB",
            lowess=True,
            line_kws={"color": "#C82B40"})
axes[1].set_title("citric acid vs volatile acidity (lowess)")

plt.tight_layout()
plt.show()
  \end{pyblock}

  \vspace{4pt}
  {\scriptsize\color{textMuted}
    \faLightbulb\ \texttt{lowess=True} ajusta una curva suave en vez de una recta.
    Útil para explorar relaciones no lineales.}
\end{frame}
